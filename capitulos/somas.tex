\subsection{EspaĂ§os $\cL^p(P)$}

Dado $p\ge 1$ arbitrĂĄrio, consideramos $\cL^p(P)$ o espaĂ§o quociente associado as variĂĄveis tais que $|X|^p$ e integrĂĄvel.

\medskip

Neste espaĂ§o reparamos que

$$\| X\|_p:= \left(E\left [ |X|^p \right]\right)^{1/p}.$$

\begin{definition}
Seja $(X_n)_{n\ge 1}$ uma sequencia de variĂĄveis aleatĂłrias.
Falaremos que uma sequĂȘncia de variĂĄveis $X_n$ converge para $X$
 em $\cL^p$ se $X_n\in \cL^p$ para todos $n\ge 1$ e que a sequencia converge pela topologia associada a norma $\| \cdot \|_p$, o de jeito equivalente se
 \begin{equation}
 \lim_{n\to \infty}  E[ |X_n-X|^p]=0.
 \end{equation}

\end{definition}


\begin{remark}
Do mesmo jeito podemos definir convergĂȘncia em $\cL^{\infty}$ o espaĂ§o de variĂĄvel associado a norma
\begin{equation}

\end{remark}

\subsection{ConvergĂȘncia em probabilidade}

\begin{definition}
Seja $(X_n)_{n\ge 1}$ uma sequencia de variĂĄveis aleatĂłrias.
Falaremos que uma sequĂȘncia de variĂĄveis $X_n$ converge para $X$ em probabilidade se para qualquer $\gep>0$

\begin{equation}
\lim_{n\to \infty} P[ |X_n-X|>\gep ]=0.
\end{equation}
\end{definition}


\begin{exercise}
A convergĂȘncia em probabilidade Ă© a convergĂȘncia induzida pela mĂ©trica
definida em $\cL^0$ por
\begin{equation}\label{l0metric}

\end{exercise}


\begin{remark}
  Note que essa noĂ§ĂŁo de convergĂȘncia em $\cL^p$ para qualquer $p>1$ implica convergĂȘncia em probabilidade.
  De fato, pela desigualdade de Markov temos
  \begin{equation}
    \lim_{n\to \infty} P[ |X_n-X|>\gep ]\le  \gep^{-p} P[ |X_n-X|^p]
  \end{equation}
\end{remark}

\subsection{ConvergĂȘncia quase certa}

Falamos que $X_n$ converge quase certamente par $X$ se
existe $\bar \gO\subset \gO$ de probabilidade $1$ tal que para todos $\go\in \bar \gO$
\begin{equation}
\lim_{n\to \infty} X_n(\go)=X(\go).
\end{equation}
O limite e Ășnico em $\cL^0$.

\begin{proposition}

\end{proposition}

\section{TĂłpico: Integrabilidade uniforme}

Vimos que convergĂȘncia em $\cL^1$ implica convergĂȘncia em probabilidade.
O contrario e claramente falso, como se pode ver com a sequĂȘncia de variĂĄveis
$$X_n:= n\1_{\{\go \in [0,n^{-1}]\}},$$
onde $\gO=[0,1]$ e $P=U([0,1])$ que converge quase certamente para $0$ mas nĂŁo converge em $\cL^1$ (Por exemplo $E[|X_n|=1]$)

\medskip

A razĂŁo da ausĂȘncia de convergĂȘncia aqui e uma concentraĂ§ĂŁo da esperanĂ§a de $|X_n|$  (poderia ser tambĂ©m sĂł uma parte dela)
em um evento cada vez menor do espaĂ§o de probabilidade ($[0,n^{-1}]$), que nĂŁo aparece no limite. Em consequĂȘncia
temos uma parte da norma $\cL^1$ que se perde no limite
(cf.\ ExercĂ­cio \ref{ex:l1}) e temos $\lim_{n\to \infty} E[X_n]$).

\medskip

Uma sequĂȘncia de variĂĄveis em $\cL^1$ que nĂŁo tem este tipo de comportamento se chama \textsl{uniformemente integral}.

\begin{definition}
 Uma coleĂ§ĂŁo de variĂĄveis aleatĂłrias $(X_i)_{i\in \cI}$ e chamada uniformemente integrĂĄvel se
 para qualquer $\gep>0$, existe $M>0$ tal que
 para todo $i\in \cI$
 \begin{equation}
  E[|X_{i}|\1_{[|X_i|>M]}]<\gep.
 \end{equation}
\end{definition}


\begin{example}
 \begin{itemize}
  \item [(a)] A coleĂ§ĂŁo $\{X\}$ que contem sĂł um elemento $X\in \cL^1$, Ă© uniformemente integrĂĄvel (e uma consequĂȘncia trivial do teorema de convergĂȘncia dominada).
  \item [(b)] Qualquer coleĂ§ĂŁo finita de variĂĄveis em $\cL^1$ Ă© uniformemente integrĂĄvel.
  \item [(c)] A bola unidade (o qualquer conjunto limitado) em $\cL^p$, $p>1$ e uniformemente integrĂĄvel.
 \end{itemize}
\end{example}

Vamos agora ver uma caracterizaĂ§ĂŁo alternativa da uniforme integrabilidade que justifica o nome Ă© a seguinte.

\begin{proposition}
  Uma coleĂ§ĂŁo de variĂĄveis aleatĂłrias $(X_i)_{i\in \cI}$  e uniformemente integrĂĄvel se
  e sĂł se para qualquer $\gep$ existe $\delta$ tal que para todos eventos  $A$ com $P(A)<\delta$ e todo $i \in \cI$
  \begin{equation}\label{eq:lacond}
    E[|X_i|\1_{A}]<\gep.
  \end{equation}
\end{proposition}

\begin{proof}
  Se a sequĂȘncia for uniformemente integrĂĄvel,
  podemos achar $M>0$ tal que para todo $n$
  \begin{equation}
    E[|X_i|\1_{[|X_i|> M]}]<\gep/2.
  \end{equation}
  Agora escolhemos $\delta=\gep/(2M)$.
  Se  $P(A)<\delta$ temos
  \begin{multline}
    E[|X_i|\1_{A}]= E[|X_i|\1_{A\cap [|X_i|\le  M]}]+E[|X_i|\1_{A\cap [|X_i|>  M]}]\\
    \le M P(A\cap [|X_i|\le  M])+ E[|X_i|\1_{[|X_i|>  M]}]\le M\delta +\gep/2\le \gep.
  \end{multline}
  Reciprocamente se \eqref{eq:lacond} for satisfeito, entĂŁo em particular
  $$ C:=\sup_n E[|X_i|]<\infty,$$
  e por desigualdade de Markov temos
  \begin{equation*}
    P[|X_i|> M] \le \frac{C}{M}.
  \end{equation*}
  EntĂŁo dado $\gep>0$, e $\delta$ tal que \eqref{eq:lacond} vale,
  podemos concluir escolhendo $M=C\delta^{-1}$ que
  $E[|X_i|\1_{[|X_i|> M]}]\le \gep$.
\end{proof}

Essa noĂ§ĂŁo permite de deduzir convergĂȘncia em $\cL^1$ de convergĂȘncia em probabilidade.

\begin{theorem}
  Seja $X_n$ uma sequĂȘncia de variĂĄveis aleatĂłrias que converge em probabilidade para $X$.
  As seguintes propriedades sĂŁo equivalentes
  \begin{itemize}
  \item [(i)] $X_n$ converge em $\cL^1$ para $X$.
  \item [(ii)] $(X_n)_{n\ge 1}$ e uniformemente integrĂĄvel.
  \end{itemize}
\end{theorem}

\begin{proof}
  Mostramos primeiro que $(i)$ implica $(ii)$.
  Primeiro a convergĂȘncia implica que dado $\gep>0$ podemos achar $N$ tal que $n\ge N$ implica
  $$E[|X_n-X_N|]\le \gep/2.$$
  Agora, o conjunto $\{X_1,\dots,X_N\}$ sendo uniformemente integrĂĄvel, existe $\delta$ tal que que
  se $P(A)<\delta$
  $$ \forall n\in \{1,\dots,N\}, \quad E[|X_n|\1_{A}]\le \gep/2.$$
  Para $n\ge N$ e $P(A)<\delta$, temos
  \begin{equation*}
    E[|X_n|\1_{A}]\le  E[|X_N|\1_{A}]+ E[|X_n-X_N|\1_{A}]\le \gep.
  \end{equation*}

  \medskip

  Agora provamos que $(ii)$ implica $(i)$.
  Notamos se $(X_n)_{n\ge 1}$ e uniformemente integrĂĄvel entĂŁo $X\in \cL_1$.
  Usando Fatou temos
  \begin{equation}
    E[|X|]\le \liminf_{n\to \infty}E[|X_n|]<\infty
  \end{equation}
  Agora reparamos que
  \begin{multline*}
    E[|X_n-X|]= E[|X_n-X| \ind_{|X_n-X|\le \gep} ]+ E[|X_n-X| \ind_{ [|X_n-X|>\gep] } ]\\
    \le
    E[|X_n| \ind_{ [|X_n-X|>\gep] } ]+E[|X|\ind_{ [|X_n-X|>\gep]}].
  \end{multline*}
  O primeiro termo e obviamente menor que $\gep$.
  Agora por consequĂȘncia da convergĂȘncia em probabilidade, a probabilidade de $[|X_n-X|>\gep]$ vai para zero.
  Por convergĂȘncia dominada, isso implica que $E[|X|\ind_{ [|X_n-X|>\gep]}]\le \gep$ para $n$ grande suficiente.

  \medskip

  Agora usando integrabilidade uniforme, temos que se $P(A)<\delta$ para $\delta$ pequeno suficiente, $E[|X_n| \ind_{A} ]<\gep$, para qualquer $n$.
  Como $P [|X_n-X|>\gep]<\delta$ para $n$ grande suficiente $E[|X_n| \ind_{ [|X_n-X|>\gep] } ]<\gep$ o que permite concluir.
\end{proof}

\begin{theorem}

\begin{proof}

Podemos extrair ma subsequĂȘncia $Y_k=X_{n(k)}$ tal que
\begin{equation}

\end{equation}
Em particular temos usando a desigualdade de Markov, temos
\begin{equation}
P(|Y_k-Y_{k+1}|\ge 2^{-k})\le 2^{-k}.
\end{equation}
Pelo Lemma de Borel-Cantelli, obtemos que a sequĂȘncia
$$Y_k=Y_1+\sum_{n=2}^k (Y_k-Y_{k-1}),$$
 converge q.c.\ Chamamos $X$ a variĂĄvel obtida no limite (vista com elemento de $\cL^0$).

 \medskip

 Claramente $Y_k$ converge em probabilidade por $X$, mas usando a desigualdade triangular obtemos que
 $$d(X_n,X)\le d(Y_{n},X_n)+d(Y_n,X).$$
 A sequĂȘncia $(X_n)_{n\ge 1}$ sendo de Cauchy o primeiro converge para zero e concluĂ­mos que
 $X_n$ converge para $X$ em probabilidade.
\end{proof}

Digamos que estamos interessados em aproximar uma variĂĄvel aleatĂłria por uma constante de forma a minimizar o erro da aproximaĂ§ĂŁo.

\end{equation}
Essa equaĂ§ĂŁo obviamente possui um Ășnico mĂ­nimo em $a = E(X)$.
Ao erro da aproximaĂ§ĂŁo acima damos o nome de variĂąncia.
Corresponde a distancia quadrada ao subespaĂ§o das constantes no espaĂ§o $\cL^2$.

\begin{definition}
  Dada uma variĂĄvel aleatĂłria $X \in \mathcal{L}^2$, definimos sua variĂąncia \index{variancia@variĂąncia} como
  \begin{equation}
    \Var(X) = E \Big( \big(X - E(X)\big)^2 \Big) = E(X^2) - \big( E(X) \big)^2.
  \end{equation}
\end{definition}


Podemos alternativamente entender a variĂąncia da seguinte maneira.
Sejam $X$ e $Y$ variĂĄveis aleatĂłrias independentes em $\mathcal{L}^2$ de mesma distribuiĂ§ĂŁo.

\begin{equation}

\end{equation}

Observe pelas definiĂ§Ă”es alternativas dadas acima que
\begin{enumerate}[\quad a)]
\item $\Var(X) \geq 0$.
\item Se $a, b\in \bbR$ $\Var (aX+b)=a^2 \Var(X).$
\item Mostre que se $X \in \cL^2$, entĂŁo $\Var(X) = 0$ se e somente se $X = a$ quase certamente.
\end{enumerate}
\end{proposition}

A seguinte aplicaĂ§ĂŁo da propridade de Markov usa a variĂąncia para  estimar o quanto uma variĂĄvel aleatĂłria se desvia de sua mĂ©dia.
\begin{proposition}[Desigualdade de Chebychev]
  Se $X \in \mathcal{L}^2$ e $a > 0$, entĂŁo
  \begin{equation}
    P [ |X - E(X)| > a] \leq \frac{\Var(X)}{a^2}.
  \end{equation}
\end{proposition}

\begin{proof}
  A desigualdade segue trivialmente da cota de Markov, ao observarmos que
  \begin{enumerate}[\quad a)]
  \item $|X - E(X)| \geq 0$.
  \item $|X - E(X)| > a$ se e somente se $|X - E(X)|^2 > a^2$.
  \item $E\big(|X - E(X)|^2\big) = E\big((X - E(X))^2\big) = \Var(X)$.
  \end{enumerate}
  Isso demostra a proposiĂ§ĂŁo.
\end{proof}

\end{equation}
Mas o caso $\cL^2$ Ă© particularmente Ăștil, como veremos posteriormente.

\begin{theorem}
  \index{Lei!Fraca dos Grandes Numeros@Fraca dos Grandes NĂșmeros}
  \label{t:lei_fraca}
  Considerando $X_1, X_2, \dots$ sĂŁo i.i.d.s em $\mathcal{L}^2(P)$, definimos
  \begin{equation}
    S_n = \sum_{i=1}^n X_i.
  \end{equation}
  Para todo $\varepsilon > 0$
  \begin{equation}
    \lim_{n \to \infty} P \Big[\Big| \frac{S_n}{n} - E(X_1)\Big| > \varepsilon \Big] = 0.
  \end{equation}
  Ou seja, $\tfrac{S_n}{n} \to E(X_1)$ em probabilidade.
\end{theorem}

\begin{proof}
  Sabemos que
  \begin{equation}
    P \Big[\Big| \frac{S_n}{n} - E(X_1)\Big| > \varepsilon \Big] \leq \frac{\Var(\tfrac{S_n}{n})}{\varepsilon^2},
  \end{equation}
  pois $E(S_n/n) = (1/n) E(X_1 + \dots + X_n) = E(X_1)$.

  Mas como $\Var(S_n/n) = (1/n^2) \Var (X_1 + \dots + X_n) = (n/n^2) \Var(X_1)$, temos o resultado.
\end{proof}

\begin{remark}
 Observe que nĂłs apenas utilizamos que as variĂĄveis $X_i$ eram independentes duas a duas.
 Isso e suficiente para mostrar que $S_n/n$ converge em $\cL^2$ e em consequĂȘncia converge em probabilidade.
\end{remark}

  \end{split}
\end{equation*}
Essas duas sequĂȘncias tem a mesma distribuiĂ§ĂŁo: Existe uma funĂ§ĂŁo $F$ tal que $M_k = F(X_1, \dots, X_k)$ e $M'_k = F(X_2, \dots, X_{k+1})$ e os vetores $(X_1, \dots, X_k)$ e $(X_2, \dots, X_{k+1})$ tem a mesma distribuiĂ§ĂŁo.
AlĂ©m disso, essas sequĂȘncias convergem de maneira crescente para $M$ e $M'$ respetivamente que sĂŁo tambĂ©m identicamente distribuidas (basta notar que $P(M \le x) = \lim_{k \to \infty} P(M_k \le x)$).

\medskip

Vamos agora tentar relacionar as sequĂȘncias $M_{k + 1}$ e $M'_k$, para explorar a invariĂąncia por translaĂ§Ă”es:
\begin{equation*}
  \begin{split}
    M_{k+1}
    & \overset{S_0 = 0}= \max \Big\{
      0, \max_{1 \leq n \leq k} (S_{n + 1} - a(n + 1))
      \Big\}\\
    & = \max(0, M'_k + X_1 - a) = M'_k - \min(M'_k, a - X_1).
  \end{split}
\end{equation*}
Usando o fato que $M_k$ e $M'_k$ tem mesma distribuiĂ§ĂŁo obtemos
\begin{equation}
  E[\min(M'_k,a-X_1)]=E[M'_{k}]-E[M_{k+1}]=E[M_{k}-M_{k+1}]\le 0.
\end{equation}
A desigualdade acima Ă© bastante contra-intuitiva, pois $a - X_1$ tem esperanĂ§a estritamente positiva e $M_k' \geq 0$ cresce para $M'$.
Mas para concluir precisamos de tomar o limite de $M'_k$.
Pelo teorema de convergĂȘncia dominada, (observando que $0 \le |\min(M'_k, a - X_1)| \le |a - X_1|$) obtemos
$$E[\min(M',a-X_1)]\le 0.$$
Agora, note que a hipĂłtese de contradiĂ§ĂŁo $P(M'=\infty)=P(M=\infty)=1$ implicaria
$E[a-X_1]\le 0$,
o que Ă© uma contradiĂ§ĂŁo pois $E(X_1) = 0$ e $a > 0$.
\end{proof}

\begin{theorem}

 $$ \lim_{n\to \infty} \frac{1}{n}\sum_{k=1}^n X_k=E[X] $$

\end{theorem}


\begin{proof}

 Usando a desigualdade de Markov temos que

Em particular  o termo a direita em \eqref{eq:hop} vai para zero quando $M\to \infty$ o que permite concluir.

 \end{proof}

\subsection{Caso de variĂĄveis independentes: Os resultados}

Vamos comeĂ§ar com um exemplo para ilustrar nossas perguntas:\

$$P[Y_n=\pm 1]=1/2.$$

\medskip

E bastante fĂĄcil reparar que temos convergĂȘncia para $p > 1$, mesmo convergĂȘncia absoluta.

\medskip

O objetivo dos resultados que presentamos agora e de poder achar uma condiĂ§ĂŁo necessĂĄria e suficiente sobre o valor de $p$ para ter convergĂȘncia.
E difĂ­cil ter uma intuiĂ§ĂŁo do resultado: sabemos que  $\sum_{n=1}^{\infty} n^{-p}$ diverge para todo $p\in (0,1)$ mas tambĂ©m que
$\sum_{n=1}^{\infty}(-1)^n n^{-p}$ converge (a seria satisfaz o teste da sĂ©rie alternada).

\medskip

Uma consequĂȘncia do resultado abaixo que no caso de sinais aleatĂłrios, a convergĂȘncia ocorre se $p > 1/2$ (veremos mais tarde que esse resultado Ă© o melhor possĂ­vel).

\begin{theorem}[Uma SĂ©rie]
  SejĂ  $(X_i)_{i\in \bbN}$ uma sequencia de variĂĄveis independente tal que $E[X_i]=0$ para todo $i$, e que satisfaz tambĂ©m
  \begin{equation}
    \sum_{i=1}^\infty \Var X_i<\infty.
  \end{equation}
  Temos entĂŁo
  \begin{equation}
    P \Big[ \sum_{i=1}^{\infty} X_i \text{ converge } \Big] = 1.
\end{equation}
\end{theorem}

\begin{theorem}[Duas Series]
  SejĂ  $(X_i)_{i\in \bbN}$ uma sequencia de variĂĄveis independente satisfaz
  \begin{equation}
    \sum_{i=1}^\infty \Var X_i<\infty.
  \end{equation}
  Temos entĂŁo
  \begin{equation}
    P[\sum_{i=1}^{\infty} X_i \text{ converge } ] =
    \begin{cases}
      1 \quad \text{ se } \sum_{i=1}^\infty E[X_i] \text{ converge },\\
      0 \quad \text{ se } \sum_{i=1}^\infty E[X_i] \text{ nĂŁo converge }.
    \end{cases}
  \end{equation}
\end{theorem}

\begin{proof}[Prova to Teorema de Duas Series]
  Vamos por enquanto assumir a validade do Teorema de Uma SĂ©rie.
  Definimos $Z_i = X_i - E[X_i]$ e usando o Teorema de Uma Serie, concluimos que $\sum_{i=1}^{\infty} Z_i$ converge com probabilidade $1$.

  \medskip

  Pois usando propriedades basicas de limites, vemos que $$\sum_{i=1}^n X_i=\sum_{i=1}^n Z_i+ \sum_{i=1}^n E[X_i]$$ converge se e sĂČ se a segunda sequĂȘncia converge.
\end{proof}

As vezes a variĂąncia pode nĂŁo ser um bom criterio para avaliar convergĂȘncia: $\Var(X_i)$ pode ser grande por causa de um evento que
acontece com pouca probabilidade. O teorema seguinte permite de tratar uns desses casos:
dado um real positivo $c>0$, definimos
\begin{equation}
X_i^c= X_i \1_{\{|X_i| \ge c\}}= \begin{cases} X_i \quad \text{ se } |X_i|\le c,\\
                                    0 \quad \text{ no caso contrĂĄrio.}
                                   \end{cases}
           \end{equation}
Chamamos essas variĂĄveis aleatĂłrias as variĂąveis troncadas.

\begin{theorem}[Tres Series]
 Sejam $c>0$ um real positivo e $(X_i)_{i\in \bbN}$ uma sequencia de variĂĄveis independente que
satisfaz
\begin{itemize}
 \item [(i)]  $\sum_{i=1}^\infty P[ |X_i|>c ]<\infty,$
 \item [(ii)]  $\sum_{i=1}^\infty E[ X^c_i]<\infty,$
 \item [(iii)]  $\sum_{i=1}^\infty \Var X^c_i<\infty.$
\end{itemize}
EntĂŁo
$$ P[\sum_{i=1}^{\infty} X_i \text{ converge } ]= 1 $$
\end{theorem}

\begin{exercise}
  Mostrar que para uma sequencia de variĂĄveis independente $X=(X_i)_{i\in \bbN}$, com as notaĂ§oes do teorema acĂ­ma,
  \begin{equation*}
    \exists c>0,\ X \text{ verifica $(i)-(ii)-(iii)$}
    \quad \Leftrightarrow \quad
    \forall c>0,\ X \text{ verifica $(i)-(ii)-(iii)$}.
  \end{equation*}
\end{exercise}

\begin{proof}[Prova to Teorema de Tres Series]
  Essa vez definimos $Y_i:= X_i \1_{\{|X_i| > c\}}$
  Temos
  \begin{equation}
    \sum_{i=1}^n X_i =  \sum_{i=1}^n X_i^c + \sum_{i=1}^n Y_i
  \end{equation}
  Usando o Teorema de Duas Series para $X_i^c$, jĂ  podemos dizer que o primeiro termo converge com probabilidade $1$.
  Quanto ao segundo termo, temos
  \begin{equation}
    \sum_{i = 1}^{\infty} P[Y_i\ne 0]=   \sum_{i=1}^{\infty} P[ X_i>c ]< \infty,
  \end{equation}
  e entĂŁo pelo Lemma de Borel Cantelli,
  \begin{equation}
    P[ Y_i \ne 0 \text{ infinitas vezes}]=0.
  \end{equation}
  o que permite de concluir.
\end{proof}

\end{exercise}

\subsection{A desigualdade de Kolmogorov}

Para provar o Teorema, vamos precisar de uma feramenta Ăłtil tambĂ©m em outros contextos.
Ă uma versĂŁo mais forte de desigualdade de Chebychev.

\medskip

\begin{theorem}[Desigualdade de Kolmogorov]
Suponhamos que $X_i$ sĂŁo variĂĄveis independentes, tais que $\Var X_i < \infty$ para todos $i$.
EntĂŁo temos
\begin{equation}
  P \Big[ \max_{1\le k\le n} |S_k-E[S_k]|>\gep \Big]
  \le \frac{\Var S_n}{\gep^2}.
\end{equation}

\end{theorem}


\begin{proof}
  Susbstituindo $X_i$ por $X_i-E[X_i]$ se for necessĂĄrio, podemos supor que a esperanĂ§a das variĂĄveis vale zero.
  Vamos decompor o evento
  \begin{equation}
    \Big[ \max_{1\le k\le n} |S_k|>\gep \Big]
  \end{equation}
  com respeito ao primeiro valor de $k$ para o qual $|S_k|>\gep$.
 \begin{equation}
   \begin{split}
     E_1 & := [|S_1|\ge \gep ],\\
     E_k & := \big[
           |S_1|<\gep;\, \dots ;\,  |S_{k-1}|<\gep\, ;  \, |S_k|\ge \gep
           \big], \quad 2\le k\le n.
   \end{split}
 \end{equation}
 Obviamente temos
 \begin{equation}
   \label{eq:1}
   \Big\{\max_{1 \le k\le n} |S_k|>\gep \Big\} = \bigcup_{k=1}^n E_k,
 \end{equation}
a uniĂŁo sendo disjunta.
 Observando que $\1_{E_k} \leq \tfrac{S_k^2}{\gep^2} \1_{E_k}$, temos
 \begin{equation}
   P \Big[
   \max_{1\le k\le n} |S_k|>\gep
   \Big]
   \le \sum_{k=1}^n P(E_k)\le  \sum_{k=1}^n E \bigg[
   \frac{S^2_k}{\gep^2}\1_{E_k}
   \bigg].
 \end{equation}
 Vamos poder concluir a prova se mostramos para todo $k$
 \begin{equation}\label{therest}
   E[S^2_k\1_{E_k}]\le E[S^2_n\1_{E_k}].
 \end{equation}
 Isso implicaria que
 \begin{equation}
   \sum_{k=1}^n E \bigg[ \frac{S^2_k}{\gep^2}\1_{E_k} \bigg]
   \le E \bigg[ \frac{S^2_n}{\gep^2}  \sum_{k=1}^n \1_{E_k} \bigg]
   = \frac{\Var(S_n)}{\gep^2}.
 \end{equation}
 Para mostrar \eqref{therest}, definimos $\Delta_{k,n}= S_n-S_k$.
 Temos
 \begin{equation}
   E \big[ S^2_n\1_{E_k} \big]
   = E \big[ (S_k + \Delta_{k,n})^2\1_{E_k} \big]
   = E[S_k^2\1_{E_k}]+2 E[ \Delta_{k,n} S_k \1_{E_k}]
     + E[ \Delta^2_{k,n}\1_{E_k}].
 \end{equation}
 O terceiro termo e obviamente positivo, entĂŁo podemos concluir se mostramos  que $E[ \Delta_{k,n} S_k \1_{E_k}]=0$.
 Lembramos que $S_k\1_{E_k}$ e uma funĂ§Ă”es das $k$ primeiras variĂĄveis e $\Delta_{k,n}$ so depende das outras $n-k$
 \begin{equation}
   S_k \1_{E_k}=f(X_1,\dots,X_k) \quad \text{ e } \quad \Delta_{k,n}=g(X_{k+1},\dots,X_{n}),
 \end{equation}
 Como os $X_i$ sĂŁo independente, concluĂ­mos que  $S_k \1_{E_k}$ e $\Delta_{k,n}$ o sĂŁo, e entĂŁo

 \begin{equation}
   E[ \Delta_{k,n} S_k \1_{E_k}]=E[\Delta_{n,k}S_k \1_{E_k}]=E[\Delta_{n,k}]E[S_k \1_{E_k}]=0.
 \end{equation}
 Isso conclui a prova do teorema.
\end{proof}

\subsection{Prova do Teorema de Uma SĂ©rie}

Uma consequĂȘncia imediata da desigualdade de Kolmogorov Ă© que se tivermos $\sum_{i=1}^{\infty} \Var X_i<\infty$ e $E[X_i]=0$, entĂŁo
\begin{equation}
  \label{e:kolmogorov_var_somavel}
  P \big[ \exists n\in \bbN,  |S_n| \ge \varepsilon \big]
  \le \frac{\sum_{i=1}^{\infty} \Var X_i}{\varepsilon^2}.
\end{equation}
Essa desigualdade Ă© obtida simplesmente tomando o limite quando $n$ vai para infinito da desigualdade de Kolmogorov.

Seja $A_m^\epsilon := [ \max_{i \geq m} |S_i - S_m| \leq \epsilon ]$.
Para mostrar que $\sum_{i} X_i$ converge, basta mostrar que essa sequĂȘncia Ă© Cauchy, ou seja, precisamos verificar que
\begin{equation}
  P \big[ \cap_{\epsilon > 0} \cup_{m \geq 1} A_m^\epsilon \big] = 1.
\end{equation}
Para tanto, basta ver que para todo $\epsilon > 0$, temos $P [\cup_{m \geq 1} A_m^\epsilon] = 1$.
Como em \eqref{e:kolmogorov_var_somavel}, temos
\begin{equation}
  P \big( (A_m^\epsilon)^c \big) \leq \frac{\sum_{i \geq m} \Var(X_i)}{\epsilon^2},
\end{equation}
que converge para zero com $m$.
Logo $\lim_m P(A_m^\epsilon) = 1$, donde $P(\cup_m A_m^\epsilon) = 1$, concluindo a prova.

\subsection{DemostraĂ§ĂŁo da LGN usando o Teorema das TrĂȘs SĂ©ries}

Antes de comeĂ§ar a prova, buscando inspiraĂ§ĂŁo no Teorema das TrĂȘs SĂ©ries, mostraremos que basta considerar versĂ”es truncadas das variĂĄveis $X_i$.
Isso Ă© feito no prĂłximo

\begin{lemma}
  \label{l:LFGN}
  Sejam $Y_i = X_i \1_{[|X_i| \leq i]}$.
  EntĂŁo, para demonstrar o Teorema~\ref{t:LFGN}, basta provar que
  \begin{equation}
    \lim_{n \to \infty}\frac{1}{n} \sum_{i=1}^n Y_i = m, \text{ $P$-quase certamente.}
  \end{equation}
\end{lemma}

\begin{proof}[Prova do Lema~\ref{l:LFGN}]
  Consideramos os eventos $A_i = [X_i \neq Y_i]$.
  Obviamente,
  \begin{equation}
  \sum_{i=1}^{\infty} P(A_i) = \sum_{i=1}^{\infty} P[|X_i| \geq i] \leq
  \int_0^\infty P[|X_1| \geq t] \d t = E\big(|X_1|) < \infty.
  \end{equation}
  Logo, pelo Lema de Borel-Cantelli, temos que $P$-quase certamente $A_i$ acontece apenas finitas vezes.
  Digamos que $A_i$ nĂŁo acontece para $i > N(\omega)$.
  Dessa forma, para qualquer $n \geq 1$,
  \begin{equation}
    \Big|\frac{1}{n}\sum_{i=1}^n (X_i - Y_i)\Big| \leq \frac{1}{n}\sum_{i=1}^n |X_i - Y_i| \leq \frac{1}{n} \sum_{i \leq N(\omega)} |X_i|,
  \end{equation}
  que converge para zero $P$-quase certamente, mostrando o resultado.
\end{proof}

O prĂłximo passo para a prova da Lei Forte dos Grandes NĂșmeros Ă© cuidar da esperanĂ§a das novas variĂĄveis $Y_i$.
\begin{lemma}
  \label{l:lim_Z_n_LFGN}
  Sejam $Z_i = Y_i - E(Y_i)$, para $i \geq 1$ como acima.
  EntĂŁo, para demostrar o Teorema~\ref{t:LFGN}, basta mostrar que
  \begin{equation}
    \label{e:lim_Z_n_LFGN}
    \lim_{n \to \infty}\frac{1}{n} \sum_{i=1}^n Z_i = 0, \text{ $P$-quase certamente.}
  \end{equation}
\end{lemma}

\begin{proof}
  Supondo a convergĂȘncia em \eqref{e:lim_Z_n_LFGN}, sabemos que
  \begin{equation}
    \lim_{n \to \infty} \frac{1}{n} \sum_{i=1}^n Y_i - E(Y_i) = 0, \text{ $P$-quase certamente.}
  \end{equation}
  Mas $E(Y_i) = E(X_i \1_{[|X_i| \leq i]})$ que converge a $E(X_i) = m$, pelo Teorema da ConvergĂȘncia Dominada, donde concluĂ­mos que
  \begin{equation}
    \lim_{n \to \infty} \frac{1}{n} \sum_{i=1}^n E(Y_i) = m.
  \end{equation}
  Dessa forma, obtemos que $\tfrac 1n \sum_{i=1}^n Y_i$ converge quase certamente a $m$, donde concluĂ­mos a prova do Teorema~\ref{t:LFGN} por meio do Lema~\ref{l:LFGN}.
\end{proof}

GostarĂ­amos de utilizar os teoremas das sĂ©ries para mostrar a convergĂȘncia de $\tfrac 1n \sum_{n} Z_n$, mas obviamente, o fator $\tfrac 1n$ que precede a soma nos impede de fazĂȘ-lo.
O prĂłximo resultado Ă© um simples exercĂ­cio de anĂĄlise real, que nos permite reduzir a prova de \eqref{e:lim_Z_n_LFGN} para uma simples convergĂȘncia de uma sĂ©rie sem prĂ©-fatores.

\begin{lemma}[Lema de Kronecker]
  Suponha que $x_n \in \mathbb{R}$ e $b_n > 0$ sejam tais que $b_n \uparrow \infty$ e $\sum_{i=1}^\infty \frac{x_i}{b_i}$ convirja a $s \in \mathbb{R}$.
  EntĂŁo
  \begin{equation}
    \lim_{n \to \infty} \frac{1}{b_n} \sum_{i=1}^n x_i = 0.
  \end{equation}
\end{lemma}

\begin{proof}
  Definindo $s_0 = 0$ e $s_n = \tfrac{x_1}{b_1} + \dots + \tfrac{x_n}{b_n}$, temos, por integraĂ§ĂŁo por partes,
  \begin{equation}
    \sum_{i=1}^n x_i = \sum_{i=1}^n b_i \frac{x_i}{b_i} = \sum_{i=1}^n b_i s_{i} - \sum_{i=1}^n b_i s_{i-1} = b_n s_n + \sum_{i=1}^{n-1} (b_{i} - b_{i+1}) s_{i}.
  \end{equation}
  Escolhemos agora, para qualquer $\varepsilon > 0$, um $n_0 \geq 1$ tal que $|s_n - s| < \varepsilon$ para todo $n \geq n_0$.
  Dessa forma,
  \begin{equation*}
    \begin{split}
      \frac{1}{b_n} \sum_{i=1}^n x_i & = s_n - \frac{1}{b_n}\sum_{i=1}^{n-1} (b_{i+1} - b_{i}) s_{i}\\
      & = s_n - \frac{1}{b_n}\underbrace{\sum_{i=1}^{n_0-1} (b_{i+1} - b_{i})}_{\Delta_{n_0}} s_{i} - \frac{1}{b_n}\sum_{i=n_0}^{n-1} (b_{i+1} - b_{i}) s_{i}\\
      & = \underbrace{s_n}_{\to s} - \underbrace{\frac{1}{b_n}\Delta_{n_0}}_{\to 0} - \underbrace{\frac{1}{b_n}\sum_{i=n_0}^{n-1} (b_{i+1} - b_i) s}_{= \tfrac{(b_n - b_{n_0})s}{b_n} \to s} - \underbrace{\frac{1}{b_n}\sum_{i=n_0}^{n-1} (b_{i+1} - b_{i}) (s_{i} - s)}_{\leq \varepsilon\tfrac{(b_n - b_{n_0})}{b_n} \leq \varepsilon},
    \end{split}
  \end{equation*}
  onde os limites indicados acima representam o que acontece quando $n \to \infty$.
  A prova segue do fato de $\varepsilon$ ter sido escolhido arbitrariamente.
\end{proof}

Estamos agora em posiĂ§ĂŁo de finalizar a
\begin{proof}[Prova do Teorema~\ref{t:LFGN}]
  De acordo com o Lema de Kronecker e o Lema~\ref{l:lim_Z_n_LFGN}, Ă© suficiente mostrar que
  \begin{equation}
    \sum_{i=1}^n \frac{Z_i}{i}, \text{ converge quase certamente}.
  \end{equation}
  Por outro lado, como os $Z_i$'s tem mĂ©dia zero, o Teorema de Uma SĂ©rie diz que Ă© suficiente mostrar que
  \begin{equation}
    \sum_{i=1}^n \Var\Big(\frac{Z_i}{i}\Big) = \sum_{i=1}^n \frac{1}{i^2} \Var(Z_i) < \infty.
  \end{equation}
  Isso segue da seguinte estimativa
  \begin{equation}
    \begin{split}
      \sum_{i=1}^n \frac{1}{i^2} \Var(Z_i) & = \sum_{i=1}^n \frac{1}{i^2} \Var(Y_i) \leq \sum_{i=1}^n \frac{1}{i^2} E\big( X_i^2 \1_{[|X_i| \leq i]}\big)\\
      & = \sum_{i=1}^n \frac{1}{i^2} \sum_{k=1}^{i} E\big( X_1^2 \1_{[k-1 < |X_1| \leq k]}\big)\\
      & = \sum_{k=1}^n E\big( X_1^2 \1_{[k-1 < |X_1| \leq k]}\big) \sum_{i=k}^{n} \frac{1}{i^2}\\
      & \leq 2 \sum_{k=1}^n \frac{1}{k} E\big( X_1^2 \1_{[k-1 < |X_1| \leq k]}\big)\\
      & \leq 2 \sum_{k=1}^n E\big( X_1 \1_{[k-1 < |X_1| \leq k]}\big) \leq 2E(X_1) < \infty.
    \end{split}
  \end{equation}
  Isso nos permite concluir a prova de \eqref{e:lim_Z_n_LFGN} via o Lema de Kronecker.
  Consequentemente, obtemos o Teorema~\ref{t:LFGN} via o Lema~\ref{l:lim_Z_n_LFGN}.
\end{proof}

\section{Momentos exponenciais}

Nessa seĂ§ĂŁo desenvolveremos uma outra tĂ©cnica para estimar a probabilidade de uma variĂĄvel aleatĂłria se desviar de sua esperanĂ§a.

JĂĄ vimos o mĂ©todo do primeiro, segundo e quarto momento para controlar uma soma de variĂĄveis independentes.
Um exemplo disso foi visto na estimativa
\begin{equation}
  P\Big[ \sum_{i=1}^n (X_i - E(X_i)) \geq a \Big] \leq \frac{\sum_i \Var (X_i)}{a^2}.
\end{equation}

Em geral, quanto maior o momento, melhor a estimativa do decaimento para a probabilidade de que uma variĂĄvel se desvie de sua esperanĂ§a.
Nessa seĂ§ĂŁo iremos para momentos exponenciais, que em um certo sentido produzem estimativas Ăłtimas para o comportamento assintĂłtico da probabilidade de desvio.

Note que se quisermos uma pequena probabilidade de erro (como por exemplo $\sim 0.01$), o mĂ©todo do segundo momento Ă© muito bom, como veremos posteriormente.
Mas se quisermos uma probabilidade de erro minĂșscula (em situaĂ§Ă”es concretas, algo como $10^{-12}$ por exemplo), certamente teremos que aumentar bastante o valor de $n$, mas quanto?
As cotas de segundo momento sĂŁo muito ruins para esse tipo de estimativa, nos levando a escolher um $n$ maior que o necessĂĄrio.
Abaixo, desenvolveremos um mĂ©todo mais eficiente para responder a essa pergunta, obviamente sob certas hipĂłteses na distribuiĂ§ĂŁo das variĂĄveis aleatĂłrias.

\begin{definition}
  Dada uma variĂĄvel aleatĂłria $X$, definimos sua transformada de Laplace \index{trasformada!de Laplace} como
  \begin{equation}
    \phi_X(s) = E(\ex{s X}) \in (0, \infty],
  \end{equation}
  para todos $s \in \mathbb{R}$.
  Essa transformada tambĂ©m Ă© chamada \emph{funĂ§ĂŁo geradora de momentos} de $X$. \index{funcao@funĂ§ĂŁo!geradora de momentos}
\end{definition}

\begin{exercise}
  Calcule a funĂ§ĂŁo geradora de momentos das distribuiĂ§Ă”es $\Ber(p)$, $\Exp(\lambda)$ e $U_{[0,1]}$.
\end{exercise}

\begin{proposition}
  \label{p:propried_phi}
  Se para algum $\delta > 0$ temos $E(\ex{\delta |X|}) < \infty$, entĂŁo
  \begin{enumerate}[\quad a)]
  \item \label{e:propried_phi_Lp} $X \in \mathcal{L}^p$ para todo $1 \leq p < \infty$,
  \item \label{e:propried_phi_finita} $\phi_X(s) < \infty$ para todo $s \in (-\delta, \delta)$,
  \item \label{e:propried_phi_suave} $\phi_X(s)$ Ă© $C^\infty$ em $(-\delta, \delta)$ e
  \item \label{e:propried_phi_momentos} $\phi_X^{(n)}(s) = E(X^n \ex{sX})$.
  \item \label{e:propried_phi_convexa} $s \mapsto \log \phi_{X}(s)$ Ă© convexa no intervalo onde estĂĄ definida.
  \end{enumerate}
\end{proposition}

A penĂșltima conclusĂŁo da proposiĂ§ĂŁo acima justifica a nomenclatura funĂ§ĂŁo geradora de momentos pois $\phi_X^{(n)}(0) = E(X^n)$.

\begin{proof}
  Obviamente, para todo $p \geq 1$ existe $c > 0$ tal que $\ex{\delta |x|} \geq c |x|^p$, donde $X \in \mathcal{L}^p$, concluindo \ref{e:propried_phi_Lp}).
  AlĂ©m disso, para todo $s \in (-\delta, \delta)$, temos $\phi_X(s) = E(\ex{s X}) \leq E(\ex{\delta |X|}) < \infty$, donde \ref{e:propried_phi_finita}) segue imediatamente.

  Fixando $s \in \mathbb{R}$, vamos agora calcular
  \begin{equation}
      \frac{\phi_X(s + h) - \phi_X(s)}{h} = \frac{E\big(\ex{(s+h)X} - \ex{sX}\big)}{h} = E\Big(\ex{sX} \frac{\ex{hX} - 1}{h}\Big).
  \end{equation}
  Lembrando que $|\tfrac{1}{y}(e^y - 1)| \leq e^{|y|}$, para todo $y \in \mathbb{R}$, temos que para todos os $h < (\delta - |s|)/2$, o integrando acima Ă© dominado por $|X| \ex{(|s| + h) |X|} \leq |X| \ex{\smash{\tfrac{\delta + |s|}{2} |X|}}$ que pertence a $\mathcal{L}^1$.
  Logo podemos usar o Teorema da ConvergĂȘncia Dominada para trocar o limite $h \to 0$ com a esperanĂ§a, obtendo
  \begin{equation}
    \phi_X'(s) = E(X \ex{sX}).
  \end{equation}

  Note que para todo $\varepsilon > 0$ e $k \geq 1$, $|x|^k \leq c(k) \ex{\varepsilon |x|}$, isso nos permite repetir o argumento acima indutivamente para obter
  \textit{c)} e \textit{d)}.

  \medskip

  O ultimo ponto e simplesmente uma consequencia da desigualdade de H\"older. Para $\gl\in (0,1)$ temos
  \begin{multline}
   \phi_X( \gl a +(1-\gl)b)= E[e^{\gl a X} e^{(1-\gl)b X}] \\
   \le E[e^{a X}]^\gl E[ e^{b X}]^{(1-\gl)}=  (\phi_X(a))^{\gl}  (\phi_X(b))^{1-\gl}
  \end{multline}
  e o resultado segue tomando o logarĂ­tmo de cada um dos lados.
\end{proof}

Lembramos que ao usar o mĂ©todo do segundo momento, nos foi bastante Ăștil o fato que a variĂąncia se comporta bem com relaĂ§ĂŁo a somas independentes.
Mais precisamente, $\Var(X_1 + \dots + X_k) = \Var(X_1) + \dots + \Var(X_k)$.

Uma outra propriedade importante da funĂ§ĂŁo geradora de momentos Ă© que ela tambĂ©m se comporta bem com respeito Ă  somas independentes.
\begin{proposition}
  Se $X_1, \dots, X_n$ sĂŁo variĂĄveis independentes com $\phi_{X_i}(s) < \infty$ para todo $i \leq k$ e $|s| < \delta$, entĂŁo
  \begin{equation}
    \phi_{X_1 + \dots + X_k}(s) = \phi_{X_1}(s) \dotsm \phi_{X_k}(s), \text{ para todos $|s| < \delta$.}
  \end{equation}
\end{proposition}

\begin{proof}
  Basta observar que, como $E[e^{s X_i}] < \infty$ para todo $i \leq k$,
  \begin{equation}
    \begin{split}
      E (\exp & \{s(X_1 + \dots + X_k)\}) = E(\ex{sX_1} \dotsm \ex{sX_k})\\
      & = E \big( \ex{sX_1} \big) \dotsm E \big( \ex{sX_k} \big) = \phi_{X_1}(s) \dotsm \phi_{X_k}(s),
    \end{split}
  \end{equation}
  usando Fubini.
\end{proof}

Consideraremos agora uma sequĂȘncia $X_1, X_2, \dots$ de variĂĄveis \iid com $\phi_{X_1}(s) < \infty$ para $|s| < \delta$.
EntĂŁo podemos tentar estimar, para $a > 0$ e $|s| < \delta$,
\begin{equation*}
  \begin{split}
    P \Big[ & \frac{X_1 + \dots + X_n}{n} - E(X_1) \geq a \Big] = P \Big[ X_1 + \dots + X_n \geq (a + E(X_1)) n \Big]\\
    & \quad = P \Big[ \ex{s(X_1 + \dots + X_n)} \geq \ex{s (a + E(X_1)) n}\Big]\\
    & \quad \leq \phi_{X_1 + \dots + X_n}(s) \ex{-s (a + E(X_1))n } = \phi_{X_1}^n(s) \ex{-s (a + E(X_1))n }.
  \end{split}
\end{equation*}
O primeiro fator na estimativa acima pode crescer exponencialmente com $n$, enquanto o segundo decresce.
GostarĂ­amos que o comportamento do segundo predominasse, o que podemos concluir do seguinte argumento.

Sabemos que $\phi_{X_1}(s)$ Ă© diferenciĂĄvel em zero e que $\phi'_{X_1}(0) = E(X_1)$.
Logo, existe $s > 0$ tal que $\phi_{X_1}(s) < 1 + (E(X_1) + \tfrac{a}{2}) s$, donde
\begin{equation*}
  \begin{split}
    P \Big[ & \frac{X_1 + \dots + X_n}{n} - E(X_1) \geq a \Big] \leq \phi_{X_1}^n(s) \ex{-s (a + E(X_1))n }\\
    & \quad \leq \big(1 + (E(X_1) + \frac{a}{2})s \big)^n \ex{-s (E(X_1) + a)n }\\
    & \quad \leq \exp\Big\{ s \Big( E(X_1 + \frac{a}{2} - E(X_1) - a) n \Big) \Big\} = \ex{-san/2}.
  \end{split}
\end{equation*}
Isso nos garante um decaimento exponencial da probabilidade da mĂ©dia dos $X_i$ se desviar da esperanĂ§a.

\begin{exercise}
  Aplique o mĂ©todo acima para variĂĄveis $X_i$ \iid com distribuiĂ§ĂŁo $\Ber(1/2)$ e encontre $s(a)$ que otimize o decaimento da probabilidade $P\big[\sum_{i=1}^n X_i > (1/2 + a) n \big]$.
\end{exercise}

PoderĂ­amos nos perguntar se a cota acima Ă© suficientemente boa.
Talvez pudĂ©ssemos esperar um decaimento ainda melhor que exponencial.
Para responder a essa pergunta, vamos considerar o seguinte exemplo.
Sejam $(X_i)_{i \geq 1}$ variĂĄveis \iid com $X_1 \distr \Ber(1/2)$.
Nesse caso temos por exemplo
\begin{equation}
  P\Big[ \big| \frac{X_1 + \dots + X_n}{n} - \frac 12 \big| \geq \frac 14\Big] \geq P[X_i = 1, \forall i \leq n] = 2^{-n}.
\end{equation}
Dessa forma, sabemos que nĂŁo podemos esperar um decaimento melhor que exponencial, mesmo para variĂĄveis bem simples (como Bernoulli) que satisfazem $\phi_X(s) < \infty$ para todo $s \in \mathbb{R}$.

Note que para variĂĄveis com distribuiĂ§ĂŁo $\Ber(1/2)$, obtivemos acima cotas exponenciais em $n$ (superior e inferior), mas elas possuem expoentes diferentes.
Resta agora tentar entender qual Ă© o expoente correto para o decaimento da probabilidade $P[X_1 + \dots + X_n \geq n(E(X_1) + a)]$, o que serĂĄ feito na prĂłxima seĂ§ĂŁo.

\begin{theorem}[PrincĂ­pio de Grandes Desvios - cota superior]

  \end{equation}
  onde $m = E(X_1)$ e
  \begin{equation}
    \psi_{X_1}(x) = \sup_{s \in \bbR} \big\{ xs - \log \big( \phi_{X_1}(s) \big) \big\}
  \end{equation}
  Ă© chamada funĂ§ĂŁo taxa. \index{funcao@funĂ§ĂŁo!taxa}

  \end{equation}
\end{theorem}

  \end{equation}

\end{remark}

Antes de buscar cotas inferiores para as probabilidades de desvio, vamos estabelecer algumas propriedades da funĂ§ĂŁo $\psi_X(b)$.
Primeiramente, quando podemos dizer que o supremo na definiĂ§ĂŁo de $\psi_X$ Ă© atingido em algum $s_{\text{max}}$?
Certamente, esse nem sempre Ă© o caso, por exemplo se $X = m$ quase certamente, entĂŁo $\phi_X(s) = e^{sm}$ e o supremo definindo $\psi_X(b)$ nĂŁo Ă© atingido se $b \neq m$.

\begin{lemma}
  \label{l:smax_PGD}
  Seja $X$ uma variĂĄvel aleatĂłria tal que $\phi_X(s) < \infty$ para todo $s \in (-\delta, \delta)$.
  Supondo $a \geq 0$ Ă© tal que $P[X > m + a] > 0$, entĂŁo existe $s_{\text{max}} \geq 0$ tal que
  \begin{equation}
    \psi_X(m + a) = (m + a)s_{\text{max}} - \log\big(\phi_X(s_\text{max})\big).
  \end{equation}
\end{lemma}

\begin{proof}
Para mostrar que o mĂĄximo Ă© atingido em um ponto, Ă© suficiente mostrar que
\begin{equation}
  \lim_{s\to \infty} (m + a)s - \log\big(\phi_X(s)\big) = -\infty,
\end{equation}
pois a funĂ§ĂŁo Ă© continua.
Por hipĂłtese, existe $x > m + a$ tal que $p = P[X \geq x] > 0$, donde $\phi_X(s) \geq p e^{sx}$ e portanto podemos concluir por que
$$(m + a)s - \log\big(\phi_X(s)\big) \leq -(x - m + a)s - \log p.$$
Isso termina a prova.
\end{proof}

\begin{lemma}
  Seja $X$ uma variĂĄvel aleatĂłria tal que $\phi_X(s) < \infty$ para todo $s \in (-\delta, \delta)$.
  EntĂŁo o conjunto onde a funĂ§ĂŁo $\psi_X(x)$ Ă© finita Ă© um intervalo, na qual $\psi_X$ Ă© convexa e portanto contĂ­nua.
\end{lemma}

\begin{proof}
  Primeiramente, supomos que $a < b$ sĂŁo tais que $\psi_X(a)$ e $\psi_X(b)$ sĂŁo finitas.
  Logo, para todo $c \in (a, b)$, temos que a funĂ§ĂŁo linear $cs$ Ă© menor ou igual a $as \vee bs$, daĂ­
  \begin{equation}
    \begin{split}
      \psi_X(c) &= \sup_{s \geq 0} \{cs - \log(\phi_X(s))\} \leq  \sup_{s \geq 0} \{(as \vee bs) - \log(\phi_X(s))\}\\
      & \leq \sup_{s \geq 0} \{as - \log(\phi_X(s))\} \vee \sup_{s \geq 0} \{bs - \log(\phi_X(s))\} < \infty.
    \end{split}
  \end{equation}
  Para mostrar que $\psi_X$ Ă© convexa, observe que $\psi_X(x)$ Ă© dada pelo supremo (para $s \geq 0$) das funĂ§Ă”es afins $x \mapsto xs - \psi_X(s)$.
  Como o supremo de funĂ§Ă”es convexas Ă© tambĂ©m convexo, obtemos o enunciado do lemma.
\end{proof}

  \begin{enumerate}[\quad a)]
  \item A funĂ§ĂŁo $\psi_{X}(s)$ Ă© nĂŁo negativa, semi-contĂ­nua inferior e convexa em seu domĂ­nio.
  \item $\psi_X(a)$ se anula somente em $a = m$.
  \end{enumerate}
\end{exercise}

Buscaremos agora cotas inferiores para a probabilidade de obter um grande desvio.
GostarĂ­amos que essas estimativas fossem o mais prĂłximas possĂ­veis das estimativas superiores obtidas acima.
Certamente nĂŁo podemos obter algo como
\begin{equation}
  \label{e:PGDgeq_falso}
  `` P\big[ X_1 + \dots + X_n \geq \big(m + a \big) n \big] \geq \exp\{-\psi_{X_1}(a) n\} ",
\end{equation}
pois senĂŁo isso nos daria uma igualdade o que Ă© impossĂ­vel, pois perdemos um pouco de precisĂŁo ao utilizar a desigualdade de Markov na cota superior.

Contudo, gostarĂ­amos de entender se ao menos o expoente $\psi_{X_1}(a)$ na cota superior tambĂ©m possui algum papel na cota inferior.
Isso Ă© confirmado no seguinte resultado.

\begin{theorem}[PrincĂ­pio de Grandes Desvios - cota inferior]
  \index{Principio de Grandes Desvios@PrincĂ­pio de Grandes Desvios}
  \label{t:PGDgeq}
  Sejam $X_1, X_2, \dots$ variĂĄveis aleatĂłrias \iid com $\phi_{X_1}(s) < \infty$, para todo $s \in \mathbb{R}$.
  EntĂŁo, para todo $a > 0$,
  \begin{equation}
    \liminf_{n \to \infty} \frac{1}{n} \log P\big[ X_1 + \dots + X_n \geq \big(m + a \big) n \big] \geq -\psi_{X_1}(m + a),
  \end{equation}
  onde novamente $m = E(X_1)$ e $\psi_{X_1}(x)$ Ă© definida como no Teorema~\ref{t:PGDleq}.
  De jeito analogo
    \begin{equation}
    \liminf_{n \to \infty} \frac{1}{n} \log P\big[ X_1 + \dots + X_n \leq \big(m - a \big) n \big] \geq -\psi_{X_1}(m - a),
  \end{equation}
\end{theorem}

Note que o resultado do teorema acima Ă© mais fraco que o que vemos na equaĂ§ĂŁo \eqref{e:PGDgeq_falso}, mas mostra que $\psi_{X_1}(a)$ Ă© realmente o expoente correto no decaimento da probabilidade de grandes desvios.

Um corolĂĄrio dos Teoremas~\ref{t:PGDleq} e \ref{t:PGDgeq} Ă© o seguinte

\begin{corollary}
  Se $X_1, X_2, \dots$ variĂĄveis aleatĂłrias \iid com $\phi_{X_1}(s) < \infty$, para todo $s \in \mathbb{R}$, entĂŁo
  \begin{equation}
    \lim_{n \to \infty} \frac{1}{n} \log P\big[ X_1 + \dots + X_n \geq \big(m + a \big) n \big] = -\psi_{X_1}(m + a).
  \end{equation}
\end{corollary}

A idĂ©ia da prova Ă© transformar a distribuiĂ§ĂŁo de $X_i$, usando uma exponencial como derivada de Radon-Nikodim.
Essa nova distribuiĂ§ĂŁo possuirĂĄ esperanĂ§a maior que $m + a$, de forma que se tomamos a mĂ©dia de variĂĄveis \iid $X'_1, \dots, X'_n$ distribuĂ­das dessa forma, obteremos algo que se concentra acima de $m + a$.
Finalmente, o preĂ§o pago para que as variĂĄveis $X_i$ se comportem como as $X'_i$ serĂĄ aproximadamente $\exp\{-\psi_{X_1}(m + a)\}$, como desejado para nossa cota inferior.

\begin{proof}
  Primeiramente, consideraremos o caso $P[X_1 \leq m + a] = 1$, que se assemelha ao caso que analizamos acima $(\Ber(1/2) \leq 1)$.
  Nesse caso, temos
  \begin{equation*}
    \begin{split}
      P\big[ X_1 + \dots + X_n \geq \big(m + a \big) n \big] & = P[X_i = m + a, \text{ para todo $i \leq n$}]\\
      & = P[X_1 = m + a]^n.
    \end{split}
  \end{equation*}
  Donde o limite acima Ă© igual a $\log(P[X_1 = m + a])$.
  Mas por outro lado,
  \begin{equation*}
    \begin{split}
      - \psi_{X_1}(m + a) & = \inf_{s \geq 0} \big\{ \log\big(E(\ex{s (X_1)})\big) - (m + a)s \big\} = \inf_{s \geq 0} \big\{ \log\big(E(\ex{s (X_1 - m - a)})\big) \big\}\\
      & \leq \liminf_{s \to \infty} \; \log\big(E(\ex{s (X_1 - m - a)})\big) = \log \big(P[X_1 = m + a]\big),
    \end{split}
  \end{equation*}
  pelo Teorema da ConvergĂȘncia Dominada, demonstrando o teorema nesse caso especial.

  Suponhamos agora que $P[X_1 > m + a] > 0$, o que implica que para $b > m + a$ suficientemente prĂłximo de $m + a$, temos $P[X_1 > b] > 0$.
  Observe que basta mostrar que para todo $b > m + a$ satisfazendo $P[X_1 > b] > 0$ e para todo $\delta > 0$, temos
  \begin{equation}
    \label{e:PGD_perto_b}
    \liminf_n \frac{1}{n} \log \Big(P\Big[\frac{X_1 + \dots + X_n}{n} \in (b - \delta, b + \delta) \Big]\Big) \geq -\psi_{X_1}(b),
  \end{equation}
  pois a funĂ§ĂŁo $\psi_{X_1}(x)$ Ă© convexa, portanto contĂ­nua.

  Vamos definir uma nova distribuiĂ§ĂŁo $\nu$ com derivada de Radon-Nikodim
  \begin{equation}
    \frac{\d \nu}{\d P_{X_1}} = \frac{1}{Z_\sigma} \ex{\sigma x}.
  \end{equation}
  Observamos primeiramente que o valor de $\sigma$ ainda nĂŁo foi escolhido.
  AlĂ©m disso apĂłs escolhido $\sigma$, teremos que calcular a constante de normalizaĂ§ĂŁo $Z_{\sigma}$ de forma que $\nu$ seja uma probabilidade.

  Escolheremos $\sigma \geq 0$ como no Lema~\ref{l:smax_PGD}, isto Ă©, tal que $\psi_{X_1}(b) = b\sigma - \log\big( \phi_{X_1}(\sigma) \big)$.
  Isso nos dĂĄ imediatamente que $Z_\sigma = E[\ex{\sigma X_1}] = \phi_{X_1}(\sigma)$
  por definiĂ§ĂŁo.

  Por diferenciabilidade de $\phi_{X_1}$, o mĂĄximo deve ser assumido em um ponto de derivada zero para a funĂ§ĂŁo $\psi_{X_1}$, ou seja
  \begin{equation}
    b = \frac{\phi_{X_1}'(\sigma)}{\phi_{X_1}(\sigma)} \overset{\text{Prop.~\ref{p:propried_phi}}}= \frac{E(X \ex{\sigma X})}{E(\ex{\sigma X})} = \frac{E(X \ex{\sigma X})}{Z_\sigma} = \int x \nu(\d x).
  \end{equation}
  Isso implica que se uma variĂĄvel aleatĂłria tem distribuiĂ§ĂŁo $\nu$, sua esperanĂ§a Ă© $b$.
  Ă possĂ­vel verificar que uma tal variĂĄvel aleatĂłria $X'$ satisfaz obrigatoriamente $\phi_{X'}(s) < \infty$ para todo $s \geq 0$, donde $X' \in \mathcal{L}^p$ para todo $p > 1$.

  Como prometido, consideramos variĂĄveis $X_1', X_2', \dots$ \iid com distribuiĂ§ĂŁo $\nu$.
  Pela lei fraca dos grandes nĂșmeros, para qualquer $\delta > 0$,
  \begin{equation}
    \lim_n P\Big[ \frac{X_1' + \dots + X_n'}{n} \in (b-\delta,b+\delta) \Big] = 1.
  \end{equation}

  Finalmente vamos relacionar essa probabilidade Ă  probabilidade definida em termos de $X_i$, na qual estamos interessados.
  \begin{equation*}
    \begin{split}
      P\Big[ & \frac{X_1 + \dots + X_n}{n} \in (b-\delta, b+\delta) \Big] = \int_{x_i; \big| \tfrac{1}{n} \sum_{i \leq n} x_i - b\big| < \delta} \;\; \bigotimes_{i=1}^n (X_1 \circ P)(\d x_i)\\
      & = Z_\sigma^n \int_{x_i; \big| \tfrac{1}{n} \sum_{i \leq n} x_i - b \big| < \delta} \;\; \ex{-\sigma \textstyle{\sum_{i=1}^n x_i}} \bigotimes_{i=1}^n \nu(\d x_i)\\[2mm]
      & \geq Z_\sigma^n \exp\{-(b + \delta) \sigma n\} P\Big[ \frac{X_1' + \dots + X_n'}{n} \in (b-\delta,b+\delta) \Big].
    \end{split}
  \end{equation*}
  Tomando o logarĂ­tmo, dividindo por $n$ e tomando o liminf quando $n$ vai a infinito, recuperamos
  \begin{equation}
    \begin{split}
      \lim_n \frac{1}{n} \log \Big(P\Big[ & \frac{X_1 + \dots + X_n}{n} \in (b - \delta,b +  \delta) \Big] \Big) \geq \log(Z_\sigma) - (b + \delta) \sigma\\
      & = \log(\phi_{X_1}(\sigma)) - (b + \delta) \sigma = -\psi_{X_1}(\sigma) - \delta \sigma.
    \end{split}
  \end{equation}
  Como isso vale para todo $\delta > 0$, provamos \eqref{e:PGD_perto_b} o que conclui a prova do teorema.
\end{proof}

\end{exercise}

\section{ConvergĂȘncia fraca}

Em muitos casos Ă© importante termos bem definida uma noĂ§ĂŁo de convergĂȘncia de medidas de probabilidade.
Supondo por exemplo no espaĂ§o mensurĂĄvel $(E,\mathcal{A})$, tenhamos uma sequĂȘncia de probabilidades $\mu_n$ e gostarĂ­amos de saber se ela converge a uma determinada $\mu$.

\begin{equation}
  d_{\VT}(\mu,\nu) = \sup_{A \in \mathcal{A}} |\mu(A) - \nu(A)|.
\end{equation}
NĂŁo Ă© difĂ­cil mostrar que a definiĂ§ĂŁo acima induz uma mĂ©trica, mas ela possui alguns problemas que descreveremos a seguir.

\begin{exercise}

\end{exercise}

  Encontre uma fĂłrmula para $d_{\VT}(\mu, \nu)$ em termos das densidades.
  Essa fĂłrmula nos remete a qual distĂąncia entre funĂ§Ă”es?
\end{exercise}

Digamos que o espaĂ§o amostral $E$ jĂĄ seja provido de uma mĂ©trica $d$ e $\mathcal{A}$ seja a $\sigma$-ĂĄlgebra dos borelianos em $E$.
Qualquer que seja a noĂ§ĂŁo de convergĂȘncia que iremos considerar, gostarĂ­amos de dizer que $\delta_{x_n}$ converge a $\delta_x$ sempre que $x_n \to x$ em $E$.

\begin{equation}

\end{equation}

Aqueles que jĂĄ viram o conceito de convergĂȘncia fraca acharĂŁo natural que a convergĂȘncia de $\mu_n$ para $\mu$ seja definida em termos da convergĂȘncia das integrais $\int f \d \mu_n$ para $\int f \d \mu$.

\begin{definition}
  Dizemos que uma sequĂȘncia de medidas de probabilidade $\mu_n$ converge fracamente (ou converge em distribuiĂ§ĂŁo) para uma probabilidade $\mu$ se \index{convergencia@convergĂȘncia!fraca}
  \begin{equation}
    \lim_{n \to \infty} \int f \d \mu_n = \int f \d \mu, \text{ para toda $f:E \to \mathbb{R}$ contĂ­nua e limitada.}
  \end{equation}
  Essa convergĂȘncia muitas vezes Ă© denotada por $\mu_n \Rightarrow \mu$.
\end{definition}

  \end{equation}
  Mostre que $M_n - \log(n)$ converge fracamente e identifique o limite.
  Observe que nĂŁo precisamos dividir $M_n - \log(n)$ por nada para obter a convergĂȘncia.
\end{exercise}

NĂłs algumas vezes denotamos $X_n \Rightarrow X$ quando $X_n$ e $X$ sĂŁo elementos aleatĂłrios de $(\Omega, \mathcal{F}, P)$ para descrever a convergĂȘncia fraca de suas respectivas distribuiĂ§Ă”es.

\subsection{Um critĂ©rio em $\mathbb{R}^d$}

Para nos provarmos resultados com convergĂȘncia fraca,
e util o resultado seguinte que fala que nĂŁo precisa verificar convergĂȘncia para todo $\varphi$ continua limitada.
Usamos a notaĂ§ĂŁo $C^b(\bbR^d)$ e $C^c(\bbR^d)$ pelos espaĂ§os de fonĂ§Ă”es continuas limitadas e continuas com suporte compacto respectivamente

\begin{proposition}\label{prop:dafunca}
As tres seguinte proposiĂ§Ă”es sĂŁo equivalentes
\begin{itemize}
 \item [(i)] $\forall \varphi \in C^b(\bbR^d), \quad \lim_{n\to \infty} \int_{\bbR^d} \varphi(x)\mu_n(\dd x)= \int_{\bbR^d} \varphi(x)\mu(\dd x).$
 \item [(ii)] $\forall \varphi \in C^c(\bbR^d), \quad \lim_{n\to \infty} \int_{\bbR^d} \varphi(x)\mu_n(\dd x)= \int_{\bbR^d} \varphi(x)\mu(\dd x).$
  \item [(iii)] $\forall \varphi \in H, \quad \lim_{n\to \infty} \int_{\bbR^d} \varphi(x)\mu_n(\dd x)= \int_{\bbR^d} \varphi(x)\mu(\dd x),$\\
  onde $H$ e um subcojunto denso de $C^c(\bbR^d)$ (pela norma de convergĂȘncia uniforme).
\end{itemize}

\end{proposition}

\begin{proof}
 As implicaĂ§Ă”es  $(i)\Rightarrow (ii) \Rightarrow (iii)$ sendo obvia, sĂČ vamos precisar mostrar $(iii)\Rightarrow (ii)$ e $(ii)\Rightarrow (i)$.

Pela primeira implicaĂ§ĂŁo, sendo $\gep$, e $\varphi \in C^c(\bbR^d)$ podemos achar $\varphi_{\gep}\in H$ tal que $\|\varphi-\varphi_{\gep} \|\le \gep$.
 Pois temos
 \begin{multline}
  \limsup_{n\to \infty} |\int \varphi \dd \mu_n-\int \varphi \dd \mu|\\
  \le \limsup |\int (\varphi-\varphi_{\gep}) \dd \mu_n|+|\int \varphi_{\gep} \dd \mu_n-\int \varphi_{\gep}\dd \mu|
  + |\int (\varphi-\varphi_{\gep})\dd \mu|\le
  2\gep.
 \end{multline}
O que permite a conclusĂŁo como $\gep$ e arbitrario.

\medskip

Para $(ii)\Rightarrow (i)$, seja $(f_k)_{k\ge 0}$ una sequencia de funĂ§oes em $C^c(\bbR^d)$ tal que
$f_k\in [0.1]$, $f_k \uparrow 1$ quando $k \to \infty$. Dado $\varphi \in C^b(\bbR^d)$ definimos $\varphi_k=f_k\varphi \in C^c(\bbR^d)$.
e entĂŁo por hypotese
$$\lim_{n\to \infty} \int \varphi_k \dd \mu_n= \int \varphi_k \dd \mu  $$
Temos
\begin{equation}
 \begin{split}
 \left |\int (\varphi-\varphi_k) \dd \mu_n \right| & \le  \| \varphi \| \left(1- \int f_k \dd \mu_n \right),\\
    \left|\int (\varphi-\varphi_k) \dd \mu\right| & \le  \| \varphi \| \left(1- \int f_k \dd \mu \right)
 \end{split}
\end{equation}
EntĂŁo temos
\begin{multline}
 \limsup_{n\to \infty} \left|\int \varphi\dd \mu_n-\int \varphi\dd \mu_n\right|\\
 \le   \| \varphi \| \left( \limsup_{n\to \infty} \left(1- \int f_k \dd \mu_n \right)
 + \left(1- \int f_k \dd \mu \right) \right) = 2 \| \varphi \| \left(1- \int f_k \dd \mu \right),
\end{multline}
mas o ultimo termo pode ser arbitrariamente pequeno se mandamos $k\to \infty$.
\end{proof}

\subsection{Um critĂ©rio em $\bbR$}

\begin{proposition}\label{prop:dafuncao}
Seja $(\mu_n)_{n\ge 0}$ uma sequĂȘncia de probabilidade em $\bbR$ e $F_n$ as funĂ§Ă”es acumuladas de distribuiĂ§ĂŁo associadas.
EntĂŁo $\mu_n$ converge fracamente para $\mu$ com  funĂ§ĂŁo acumulada de distribuiĂ§ĂŁo $F$ se e sĂł se
\begin{equation}\label{eq:acumulex}
 \lim_{n\to \infty} F_n(x):= F(x).
\end{equation}

para todo $x\in \bbR$ onde $F$ Ă© continua (para todo $x$ tal que $\mu(\{x\})=0$.
\end{proposition}


\begin{proof}
Vamos comeĂ§ar mostrando que \eqref{eq:acumulex} implica convergĂȘncia fraca.
Usando ProposiĂ§ĂŁo \ref{prop:dafunca}, e suficiente mostrar $\int f(x) \mu_n(\dd x)$ converge para $f$ diferenciĂĄvel com suporte compacto.
Lembramos (ProposiĂ§ĂŁo \ref{p:espera_acumulada}) que temos
 $$\int f(x) \mu_n(\dd x)= \int f'(x) (1-F_n(x)) \dd x.$$
 Como $f'$ e limitada e dado que $F(x)$ e descontinua sĂł em conjunto de medida $0$, podemos concluir por convergĂȘncia dominada que

 $$\lim_{n\to \infty} \int f'(x) (1-F_n(x)) \dd x=\int f'(x) (1-F_n(x)) \dd x=\int f(x) \mu(\dd x).$$

Reciprocamente, se $\mu_n$ converge fracamente para $\mu$, consideramos $x$ um ponto de continuidade de $F$, $\gep>0$ e
$\delta$ tal que
$$F(x)-(\gep/2) \le F(x-\delta) \le F(x+\delta)\le F(x)+(\gep/2).$$

Consideramos as funĂ§Ă”es
\begin{equation}
 g^+(u):=\begin{cases}
          1 &\text{ if } u\le x,\\
          1-\delta^{-1}(u-x) \quad &\text{ if } u\in [x,x+\delta],\\
          0 &\text{ if } u \ge x+\delta].
         \end{cases}
\end{equation}
e $g^-(u)=g^{+}(u+\delta)$. Como $\ind_{\{u\le x\}} \le  g^+(u) \le  \ind_{\{u\le x+\delta\}},$
temos
\begin{equation*}
F_n(x)\le \int g^{+}(u) \mu_n(\dd x)\le \int g^{+}(u) \mu(\dd x)+(\gep/2) \le F(x+\delta)+\gep/2\le F(x)+\gep,
\end{equation*}
 onde a segunda desigualdade pode se deduzir da convergĂȘncia fraca e vale para $n$ grande suficiente.
Usando $g^-$ podemos provar do mesmo jeito que $F_n(u)\ge F(x)-\gep$, o que acaba a demostraĂ§ĂŁo $\gep$ sendo arbitrĂĄrio.
\end{proof}

  \section{TĂłpico: O Teorema de Portmanteau}

  O prĂłximo resultado Ă© bastante Ăștil para provar convergĂȘncia fraca, pois nos fornece uma coleĂ§ĂŁo de equivalĂȘncias muitas vezes mais fĂĄceis de verificar.

  \begin{theorem}[Teorema de Portmanteau]
    \index{Teorema!de Portmanteau}
    \label{t:portmanteau}
    Sejam $(\mu_n)_{n \geq 1}$ e $\mu$ medidas de probabilidade em $(E, \mathcal{A})$.
    SĂŁo equivalentes:
    \begin{enumerate}[\quad a)]
    \item[a)] $\mu_n \Rightarrow \mu$,
    \item[a')] $\int f \d \mu_n \to \int f \d \mu$, para toda $f$ unifmormemente contĂ­nua e limitada,
    \item[b)] $\limsup_n \mu_n(F) \leq \mu(F),$ para todo $F \subseteq E$ fechado,
    \item[b')] $\liminf_n \mu_n(G) \geq \mu(G),$ para todo $F \subseteq E$ aberto,
    \item[c)] $\lim_n \mu_n(A) = \mu(A),$ para todo $A \in \mathcal{A}$ com $\mu(\partial A) = 0$.
    \end{enumerate}
  \end{theorem}

  Para memorizar o teorema acima, Ă© conveniente lembrar dos dois exemplos:
  \begin{enumerate}[\quad i)]
    \item se $x_n \to x$ com $x_n \neq x$, $F = \{x\}$ e $G = B(x, \delta) \setminus \{x\}$ temos, para $n$ grande,
      \begin{equation}
        \mu_n(F) = \mu(G) = 0 < 1 = \mu(F) = \mu_n(G),
      \end{equation}
    \item em $(\mathbb{R},\mathcal{B}(\mathbb{R}))$, seja $\mu_{2n} = \delta_n$ e $\mu_{2n+1} = \mu = \delta_0$.
      Obviamente $\mu_n$ nĂŁo converge fracamente a $\mu$. Contudo, para todo $A \in \mathcal{B}(\mathbb{R})$,
      \begin{equation}
        \begin{split}
          \liminf_n \mu_n (A) & \leq \liminf_n \mu_{2n}(A) = \mu(A) \text{ e}\\
          \limsup_n \mu_n (A) & \geq \limsup_n \mu_{2n}(A) = \mu(A).
        \end{split}
      \end{equation}
  \end{enumerate}

  \begin{proof}[Prova do Teorema~\ref{t:portmanteau}]
    Obviamente, $(a \Rightarrow a')$, pois $a')$ somente supĂ”e a convergĂȘncia das integrais para funĂ§Ă”es $f$ que sejam uniformemente contĂ­nuas, portanto Ă© um requisito mais fraco que $a)$.

    Observamos tambĂ©m que $(b \Leftrightarrow b')$.
    De fato, basta tomarmos complementos e observar a mudanĂ§a nos sinais das desigualdades.

    EntĂŁo, para a prova do teorema, basta mostrar que $(a' \Rightarrow b)$, $(b + b' \Rightarrow c)$ e $(c \Rightarrow a)$.

    ComeĂ§amos com $(a' \Rightarrow b)$ e para tanto, consideramos $F \subseteq E$ fechado.
    Seja $\delta > 0$ e defina a funĂ§ĂŁo $f_\delta: E \to \mathbb{R}$ dada por
    \begin{equation}
      f_\delta (x) = \max \Big\{ 1 - \frac{d(x, F)}{\delta}, 0 \Big\}.
    \end{equation}
    Claramente, $f$ Ă© uniformemente contĂ­nua e vale $\1{F} \leq f_\delta \leq \1{B(F,\delta)}$.
    Dessa desigualdade, temos $\limsup_n \mu_n(F) \leq \limsup_n \int f_\delta \d \mu_n = \int f_\delta \d \mu \leq \mu(B(F,\delta))$.
    Tomando agora o limite com $\delta \to 0$, obtemos $b)$ por continuidade da probabilidade $\mu$.

    Para mostrar $(b + b' \Rightarrow c)$, seja $A \in \mathcal{A}$ tal que $\mu(\partial A) = 0$.
    Nesse caso, sabemos que
    \begin{equation*}
      \begin{split}
        \limsup_n \mu_n(A) & \leq \limsup_n \mu_n(\bar A) \leq \mu (\bar A) = \mu (\mathring{A})\\
        & \leq \liminf \mu_n (\mathring{A}) \leq \liminf_n \mu_n (A),
      \end{split}
    \end{equation*}
    o que mostra o limite em $c)$.

    Finalmente, resta mostrar $(c \Rightarrow a)$ e, para tanto, consideramos uma funĂ§ĂŁo $f: E \to \mathbb{R}$ contĂ­nua e limitada.
    Digamos, com $\lVert f \rVert_\infty = M$.

    Sabemos que os conjuntos $\{f^{-1}(\{a\})\}_{a \in \mathbb{R}}$ sĂŁo disjuntos, logo os conjuntos $f^{-1}(\{a\})$ podem ter medida $\mu$ positiva apenas para uma coleĂ§ĂŁo enumerĂĄvel de valores $a \in \mathbb{R}$.
    Obtemos assim uma coleĂ§ĂŁo finita $b_0 < b_1 < \dots < b_k$, tal que
    \begin{equation}
      \begin{array}{c}

        \mu\big(f^{-1} (\{b_i\}) \big) = 0 \text{ para todo $i \leq k$}.
      \end{array}
    \end{equation}

    Obviamente $f_\delta \leq f \leq f_\delta + \delta$, donde
    \begin{equation*}

\section{A distribuiĂ§ĂŁo normal}

Nosso prĂłximo passo serĂĄ estudar qual poderia ser uma possĂ­vel forma limite para o histograma da Figura~\ref{f:histograma_normal}.
Essa seĂ§ĂŁo Ă© bastante heurĂ­stica e faz vĂĄrias suposiĂ§Ă”es nĂŁo rigorosas atĂ© que um candidato para o limite seja obtido.

Como uma primeira tentativa, suponha que $\sum_{i=1}^\infty Z_i$ possui uma certa distribuiĂ§ĂŁo $\mu$ (veremos posteriormente que isso somente pode acontecer em casos triviais).
Mas se esse fosse o caso, poderĂ­amos dividir a soma nos termos pares e Ă­mpares $X = \sum_{i \text{ par}} Z_i$ e $Y = \sum_{i \text{ Ă­mpar}} Z_i$.
Nesse caso terĂ­amos $X$ e $Y$ independentes e tambĂ©m distribuĂ­dos como $\mu$ (pois sĂŁo dados por uma soma que tem a mesma distribuiĂ§ĂŁo daquela que define $\mu$).

O seguinte lema mostra que isso somente pode acontecer na situaĂ§ĂŁo trivial em que $\mu = \delta_0$.

\begin{lemma}
  Sejam $X$ e $Y$ variĂĄveis aleatĂłrias em $\mathcal{L}^2$, \iid com distribuiĂ§ĂŁo $\mu$.
  Nesse caso, se $X + Y$ tambĂ©m tem distribuiĂ§ĂŁo $\mu$, entĂŁo $\mu = \delta_0$.
\end{lemma}

\begin{proof}
  Sabemos que
  \begin{equation}
    \begin{split}
      E(X + Y) & = E(X) + E(Y) = 2 E(X) \text{ e}\\
      \Var(X + Y) & = \Var(X) + \Var(Y) = 2 \Var(X).
    \end{split}
  \end{equation}
  Mas como $X + Y$ tem a mesma distribuiĂ§ĂŁo de $X$, entĂŁo $E(X) = 2 E(X)$ e $\Var(X) = 2 \Var(X)$, donde ambas sĂŁo zero.
  Usando o mĂ©todo dos segundo momento, para todo $a > 0$,
  \begin{equation}
    P[|X| \geq a] \leq \frac{\Var(X)}{a^2} = 0,
  \end{equation}
  terminando a prova de que $X = 0$ quase certamente.
\end{proof}

A intuiĂ§ĂŁo dessa prova Ă© que quando somamos duas variĂĄveis nĂŁo determinĂ­sticas, a incerteza da soma (medida atravĂ©s da variĂąncia) tende a aumentar.
Dessa forma nĂŁo podemos obter a mesma distribuiĂ§ĂŁo apĂłs a soma.

Mas existe uma maneira simples de tornar esse problema interessante novamente.
Digamos que $X$ e $Y$ pertencem a $\mathcal{L}^2$ e sĂŁo i.i.d.
EntĂŁo
\begin{equation}
  \Var\Big(\frac{X + Y}{\sqrt{2}}\Big) = 2 \Var\Big( \frac X{\sqrt{2}} \Big) = \Var(X).
\end{equation}
EntĂŁo podemos nos perguntar se

\begin{question}
  \label{q:ponto_fixo_soma}
  Existe alguma distribuiĂ§ĂŁo nĂŁo trivial $\mu$ em $\mathcal{L}^2$ tal que, se $X$ e $Y$ sĂŁo independentes e distribuĂ­das de acordo com $\mu$, temos
  \begin{equation}
    \frac{X + Y}{\sqrt{2}} \distr \mu \; ?
  \end{equation}
  Pelo menos sabemos agora que a variĂąncia nĂŁo se altera atravĂ©s dessa operaĂ§ĂŁo.
\end{question}

Ou em outras palavras, queremos saber se existe algum ponto fixo para o operador $\Gamma$ que toma uma distribuiĂ§ĂŁo $\mu$ em $\mathbb{R}$ e retorna
\begin{equation}
  \label{e:Gamma_operador}
  \Gamma(\mu) = \Big( \frac{X_1 + X_2}{\sqrt{2}} \Big) \circ \mu \otimes \mu.
\end{equation}


Para tentar responder a essa questĂŁo, vamos estudar mais a fundo qual Ă© a distribuiĂ§ĂŁo da soma de duas variĂĄveis aleatĂłrias independentes.
Para isso, considere a distribuiĂ§ĂŁo $(X,Y) \circ P$ do par, que coincide com $\mu \otimes \mu$, nos dando
\begin{equation}
  P\Big[ \frac{X + Y}{\sqrt{2}} \leq z \Big] = \mu \otimes \mu \big( \big\{(x, y); \tfrac{x + y}{\sqrt{2}} \leq z \big\} \big).
\end{equation}

Note tambĂ©m que a transformaĂ§ĂŁo linear $(x,y) \mapsto \tfrac{1}{\sqrt{2}}\big(x + y, x - y\big)$ Ă© uma rotaĂ§ĂŁo rĂ­gida em $\mathbb{R}^2$, o que nos motiva a propor a pergunta mais simples.

\begin{question}
  Existe alguma distribuiĂ§ĂŁo nĂŁo trivial $\mu$ em $\mathcal{L}^2$ tal que, se $X$ e $Y$ sĂŁo independentes e distribuĂ­das de acordo com $\mu$, a distribuiĂ§ĂŁo do par $(X,Y)$ Ă© invariante por rotaĂ§Ă”es?
\end{question}

Ainda estamos numa busca nĂŁo rigorosa de tal distribuiĂ§ĂŁo, entĂŁo vamos supor algumas outras propriedades, como por exemplo que $\mu$ seja absolutamente contĂ­nua com respeito a Lebesgue, isto Ă© $\d \mu = f(x) \d x$.
Nesse caso, jĂĄ vimos que $(X, Y) \distr f(x) f(y) \d x \d y$ e no fundo estamos procurando uma funĂ§ĂŁo $f$ tal que
\begin{equation}
  f(x) f(y) = h(x^2 + y^2), \text{ para todo $x, y \in \mathbb{R}$ e alguma $h: \mathbb{R}_+ \to \mathbb{R}_+$.}
\end{equation}
Para trasformar o produto $f(x) f(y)$ em uma soma, definimos $g = \log f$ e $k = \log h$ e o que gostarĂ­amos que acontecesse Ă© $g(x) + g(y) = k(x^2 + y^2)$.
Como ainda nĂŁo estamos preocupados com unicidade de $\mu$ e apenas com a existĂȘncia, jĂĄ podemos encontrar nossa resposta para nossa pergunta, escolhendo uma funĂ§ĂŁo quadrĂĄtica, tal como $g(x) = \alpha x^2 - \beta$.

Mas temos ainda que cuidar para que $f(x) = \ex{\alpha x^2 - \beta}$ seja uma densidade, ou seja $\int f \d x = 1$.
Para isso, precisamos que $\alpha$ seja negativo e, fixado $\alpha$, o valor de $\beta$ jĂĄ estarĂĄ determinado por normalizaĂ§ĂŁo.
Tudo isso motiva finalmente a seguinte definiĂ§ĂŁo.

\begin{definition}
  Dizemos que $X$ tem distibuiĂ§ĂŁo normal canĂŽnica, se \index{distribuicao@distribuiĂ§ĂŁo!normal}
  \begin{equation}
    \label{e:normal_canonica}
    X \distr \frac{1}{\sqrt{2 \pi}} \exp \big\{-x^2/2\big\} \d x.
  \end{equation}
  AlĂ©m disso, para $m \in \mathbb{R}$ e $\sigma \geq 0$, dizemos que $Y \distr \mathcal{N}(m, \sigma^2)$ se $Y$ tem a mesma distribuiĂ§ĂŁo de $\sigma X + m$, onde $X$ tem distribuiĂ§ĂŁo normal canĂŽnica $\mathcal{N}(0, 1)$. Note que $\mathcal{N}(m, 0) = \delta_m$.
  Muitas vezes chamamos essa distribuiĂ§ĂŁo de gaussiana, obviamente em homenagem a Gauss.
\end{definition}


Vamos rapidamente observar que a definiĂ§ĂŁo acima realmente descreve uma distribuiĂ§ĂŁo de probabilidade, ou seja que a integral dessa densidade Ă© um.
Para tanto, vamos usar um truque conhecido, que consiste em retornar ao plano.
Obviamente,
\begin{equation}
  \begin{split}
    \Big(\int \exp \big\{-x^2/2\big\} \d x\Big)^2 & = \int \int \exp \big\{-(x^2 + y^2)/2\big\} \d x \d y\\
    & = \int_0^{2 \pi} \int_0^\infty \exp \{ - r^2 / 2 \} r \d r \d \theta \overset{2 s \; = \; r^2}= 2 \pi.
  \end{split}
\end{equation}
Donde a constante em \eqref{e:normal_canonica} estĂĄ de fato correta.

  \begin{equation}

  \end{equation}
\end{exercise}

\end{exercise}

Para confirmar que de fato as distribuiĂ§Ă”es normais se comportam bem com respeito a somas independentes, apresentamos o seguinte resultado.

\begin{proposition}
  \label{p:soma_normais}
  Se $X \distr \mathcal{N}(m, \sigma^2)$ e $Y \distr \mathcal{N}(\bar{m}, \bar{\sigma}^2)$ sĂŁo independentes, entĂŁo $X + Y$ tem distribuiĂ§ĂŁo $\mathcal{N}(m + \bar{m}, \sigma^2 + \bar{\sigma}^2)$.
  Em particular, $\mu$ Ă© um ponto fixo do operador $\Gamma$ definido em \eqref{e:Gamma_operador}.
\end{proposition}

\begin{proof}
  O caso em que $\sigma$ ou $\bar{\sigma}$ se anulam Ă© trivial, portanto vamos considerar que ambas sĂŁo positivas.
  NĂŁo Ă© difĂ­cil ver que podemos tambĂ©m supor que $m = \bar{m} = 0$.
  Podemos entĂŁo calcular
  \begin{equation}
    P[X + Y \leq a] = P[\sigma W + \bar{\sigma} Z \leq a],
  \end{equation}
  onde $W$ e $Z$ sĂŁo independentes com distribuiĂ§ĂŁo $\mathcal{N}(0,1)$.
  Assim, a probabilidade acima pode ser escrita como
  \begin{equation}
    \label{e:soma_normal}
    \mathcal{N}(0,1) \otimes \mathcal{N}(0,1) \Big( \big\{ (w,z) \in \mathbb{R}^2; \sigma w + \bar{\sigma} z \leq a \big\} \Big).
  \end{equation}
  Agora aplicaremos a rotaĂ§ĂŁo rĂ­gida $A: \mathbb{R}^2 \to \mathbb{R}^2$ dada por
  \begin{equation}
    A(w,z) = \frac{1}{\sqrt{\sigma^2 + \bar{\sigma}^2}} \big( \sigma w + \bar{\sigma} z, \bar{\sigma} w - \sigma z \big).
  \end{equation}

  Como sabemos que a densidade $f$ de $(W,Z)$ Ă© invariante por $A$, ou seja $f \circ A = f$, entĂŁo podemos escrever \eqref{e:soma_normal} como
  \begin{equation*}
    \begin{split}
      \mathcal{N}(0,1) & \otimes \mathcal{N}(0,1) \Big( A \big(\big\{ (w,z) \in \mathbb{R}^2; \sigma w + \bar{\sigma} z \leq a \big\} \big) \Big)\\
      & = \mathcal{N}(0,1) \otimes \mathcal{N}(0,1) \Big( \Big\{(w,z); \frac{1}{\sqrt{\sigma^2 + \bar{\sigma}^2}}w \leq a \Big\} \Big)\\
      & = \mathcal{N}(0,1) \big( (-\infty, a \sqrt{\sigma^2 + \bar{\sigma}^2} \big] \big) = \mathcal{N}(0,\sigma^2 + \bar{\sigma}^2) \big( (-\infty, a \big] \big),
    \end{split}
  \end{equation*}
  terminando a prova da proposiĂ§ĂŁo.
\end{proof}

Podemos obter um corolĂĄrio interessante sobre a soma de normais i.i.d.
\begin{corollary}
  \label{c:normaliz_normais}
  Sejam $X_1, X_2, \dots$ variĂĄveis \iid com distribuiĂ§ĂŁo $\mathcal{N}(m,\sigma^2)$, entĂŁo
  \begin{equation}
    X_1 + \dots + X_n \distr \mathcal{N}(nm, n \sigma^2).
  \end{equation}
  Como consequĂȘncia
  \begin{equation}
    \frac{\sum_{i=1}^n X_i - n E(X_1)}{\sigma \sqrt{n}} \distr \mathcal{N}(0,1).
  \end{equation}
\end{corollary}

Lembrando da Lei dos Grandes NĂșmeros, se dividimos a soma dos $X_i - E(X_i)$ por $n$, essa fraĂ§ĂŁo vai a zero quase certamente.
O que concluĂ­mos acima Ă© que ao dividir por $\sqrt{n}$ obtemos um limite nĂŁo trivial (nem zero, nem infinito) e aleatĂłrio (nĂŁo determinĂ­stico).

Mais uma observaĂ§ĂŁo curiosa: nossa motivaĂ§ĂŁo para a definiĂ§ĂŁo da distribuiĂ§ĂŁo normal passou por invariĂąncia por rotaĂ§Ă”es e podemos extender essa invariĂąncia para $n$ normais independentes.
Note que somar as coordenadas canĂŽnicas Ă© equivalente a tomar o produdo escalar com o vetor $(1,1,\dots,1)$, que tem norma euclideana $\sqrt{n}$.

Uma outra maneira de entender o corolĂĄrio acima Ă© que a normal Ă© um ponto fixo da operaĂ§ĂŁo seguinte
\begin{enumerate}[\quad a)]
\item tome uma distribuiĂ§ĂŁo $\mu \in \mathcal{L}^2$,
\item considere $X_1, \dots, X_n$ \iid com distribuiĂ§ĂŁo $\mu$ e
\item retorne a distribuiĂ§ĂŁo de
  \begin{equation}
    \frac{X_1 + \dots + X_n - n E(X_1)}{\sqrt{n}}.
  \end{equation}
\end{enumerate}

Na QuestĂŁo~\ref{q:ponto_fixo_soma}, nos perguntamos quais seriam os outros possĂ­veis pontos fixos de $\Gamma$ e isso serĂĄ considerado depois.
Mas uma outra questĂŁo bastante importante Ă© se o ponto fixo $\mathcal{N}(0,1)$ Ă© atrator, ou seja se comeĂ§ando com outras distribuiĂ§Ă”es poderĂ­amos nos aproximar de $\mathcal{N}(0,1)$ Ă  medida que iteramos $\Gamma$.

Isso Ă© estudado no Teorema Central do Limite (TCL) que provaremos posteriormente.
Mas antes, precisamos desenvolver uma boa definiĂ§ĂŁo de convergĂȘncia para distribuiĂ§Ă”es, ou em outras palavras definir uma topologia.
Esse serĂĄ o nosso prĂłximo tĂłpico.

\section{O TCL para uma sequĂȘncia i.i.d.}

\begin{theorem}[Teorema Central do Limite]
  \index{Teorema!Central do Limite}
  \label{t:tcl_iid}
  Considere em $(\Omega, \mathcal{F}, P)$, uma sequĂȘncia $X_1, X_2, \dots$ de variĂĄveis aleatĂłrias \iid em $\mathcal{L}^3$.
  Nesse caso, se definimos $m = E(X_1)$ e $\sigma^2 = \Var(X_1)$, temos
  \begin{equation}
    \frac{\sum_{i=1}^n (X_i - m)}{\sigma \sqrt{n}} \Rightarrow \mathcal{N}(0,1).
  \end{equation}
\end{theorem}

\begin{proof}
  Primeiramente, observe que podemos supor que $m = 0$, pois de qualquer forma iremos subtrair a mĂ©dia da distribuiĂ§ĂŁo na qual nos interessamos.
  Uma outra observaĂ§ĂŁo importante Ă© que podemos supor $\sigma = 1$, pois no caso geral de qualquer forma estamos somando $X_i/\sigma$ no enunciado.

  Como vimos na ProposiĂ§ĂŁo~\ref{p:conv_distr_suave}, basta mostrar a convergĂȘncia das integrais de funĂ§Ă”es $g \in C^3$, que possuam todas as trĂȘs primeiras derivadas limitadas.
  Considerando a funĂ§ĂŁo
  \begin{equation}
    \phi^n(x_1, \dots, x_n) := g\Big(\frac{x_1 + \dots + x_n}{\sqrt{n}} \Big),
  \end{equation}
  nos basta provar a convergĂȘncia das sequĂȘncias de nĂșmeros reais
  \begin{equation}
    \label{e:tcl_limite_phi}
    \lim_n \int \phi^n(X_1, \dots, X_n) \d P = \int g(s) \mathcal{N}(0,1)(\d s).
  \end{equation}

  Vale lembrar que no CorolĂĄrio~\ref{c:normaliz_normais} jĂĄ estabelecemos algo mais forte para variĂĄveis normais.
  Mais precisamente, suponha que extendemos nosso espaĂ§o de probabilidade para $(\Omega', \mathcal{F}', P')$, onde exista uma sequĂȘncia $Y_1, Y_2, \dots$ de variĂĄveis aleatĂłrias \iid com distribuiĂ§ĂŁo $\mathcal{N}(0,1)$ independente de $X_1, X_2, \dots$
  EntĂŁo, para todo $n \geq 1$,
  \begin{equation}
    \int \phi^n(Y_1, \dots, Y_n) \d P' = \int g(s) \mathcal{N}(0,1) (\d s),
  \end{equation}
  o que tornaria o limite em \eqref{e:tcl_limite_phi} trivial para tais variĂĄveis.
  A nossa estratĂ©gia serĂĄ aproximar $\phi^n(X_1, \dots, X_n)$ por $\phi(Y_1, \dots, Y_n)$, e faremos isso trocando uma variĂĄvel de cada vez.

  Para entender o que acontece quando trocamos uma das variĂĄveis $X_i$ por $Y_i$, temos que expandir $g$ em sĂ©rie de potĂȘncias, isto Ă©, escrever
  \begin{equation}
    g(s) = g(s_0) + g'(s_0)(s - s_0) + g''(s_o)(s-s_0)^2/2 + r_{s_0}(s - s_0),
  \end{equation}
  onde $r_{s_0}(h)/h^3$ Ă© limitada por $M$, uniformemente em $h$ e $s_0$ em consequĂȘncia das nossas suposiĂ§Ă”es sobre $g$.

  Denotando $z_i = (y_1, \dots, y_{i-1}, x_i, \dots x_n)$, $z_i^o := (y_1, \dots, y_{n-1}, 0, x_{n+1}, \dots, x_n)$ e $s_i^o = y_1 + \dots + y_{n-1} + x_{n+1} + \dots x_n$, temos
  \begin{equation}
    \phi^n(z_i) %& = \phi^n(z_i^o) + \frac{\partial \phi^n}{\partial x_i} (z_i^o) x_i + \frac{\partial^2 \phi^n}{\partial x_i^2} (z_i^o) \frac{x_i^2}{2} + r_z(x_i)\\
    = \phi^n(z_i^o) + g' \Big( \frac{s_i^o}{\sqrt{n}} \Big) \frac{x_i}{\sqrt{n}} + g'' \Big( \frac{s_i^o}{\sqrt{n}} \Big) \frac{x_i^2}{2n} + r_{\frac{s_i^o}{\sqrt{n}}} \Big( \frac{x_i}{\sqrt{n}} \Big),
  \end{equation}
  NĂłs propositalmente expandimos $\phi^n$ atĂ© ordem dois, pois $X_i$ e $Y_i$ possuem os mesmos momentos de ordem um ($m=0$) e dois ($\sigma^2=1$).

  Integrando os dois lados da igualdade acima com respeito a $Z_i \circ P$ (denotamos como antes, $Z_i = (Y_1, \dots, Y_{i-1}, X_i, \dots, X_n)$ e $Z_i^o$, $S_i^o$ analogamente), teremos
  \begin{equation}
    \int \phi^n(Z_i) \d P' = \int \phi^n(Z_i^o) \d P' + \frac{1}{2n} v_i + k_i,
  \end{equation}
  onde as quantidades $v$ e $k$, se escrevem como
  \begin{equation}
    v_i = \int g'' \Big( \frac{S_i^o}{\sqrt{n}} \Big) \d P' \quad \text{ e } \quad k_i = \int r_{S_i^o/\sqrt{n}} \Big(\frac{X_i}{\sqrt{n}}\Big) \d P'.
  \end{equation}
  Note que $v_i$ nĂŁo depende de $X_i$ e que
  \begin{equation}
    |k_i| \leq \Big| \int \Big(\frac{X_i^3}{n^{3/2}}\Big) \Big(\frac{n^{3/2}}{X_i^3}\Big) r_{S_i^o/\sqrt{n}} \Big(\frac{X_i}{\sqrt{n}}\Big) \d P' \Big| \leq \frac{M}{n^{3/2}} E(|X_i^3|).
  \end{equation}

  As observaĂ§Ă”es acima sĂŁo o ponto mais importante da prova de que essa aproximaĂ§ĂŁo funciona e uma outra maneira de colocĂĄ-las Ă© a seguinte.
  Como $X_i$ e $Y_i$ possuem os dois primeiros momentos iguais, os dois primeiros termos de Taylor coincidem apĂłs a integraĂ§ĂŁo (o primeiro se anula e o segundo Ă© $v_i$ tanto para $X_i$ quanto para $Y_i$).
  O resto Ă© de ordem muito pequena para influir no limite.

  De fato, se retiramos o termo $Y_i$ de $Z_{i+1}$, fazendo a mesma expansĂŁo que para $X_i$, obtemos
  \begin{equation}
    \int \phi^n(Z_{i+1}) \d P' = \int \phi^n(Z_i^o) \d P' + \frac{1}{2n} v_i + k'_i,
  \end{equation}
  com o termo de ordem superior $k'_i$ sendo definido exatamente como $k_i$, mas com $Y_i$ no lugar de $X_i$.

  Estamos prontos agora para a computaĂ§ĂŁo final
  \begin{equation*}
    \begin{split}
      \Big| \int \phi^n & (X_1, \dots, X_n) \d P - \int g(s) \mathcal{N}(0,1)(\d s) \Big|\\
      & = \Big| \int \phi^n(Z_0) \d P' - \int \phi^n(Z_n) \d P' \Big|\\
      & \leq \sum_{i=0}^{n-1} \Big| \int \phi^n(Z_{i}) \d P' - \int \phi^n(Z_{i+1}) \d P' \Big| = \sum_{i=0}^{n-1} |k_i - k'_i|\\
      & \leq n \frac{M}{n^{3/2}} \big(E(|X_1|^3) + E(|Y_1|^3) \big),
    \end{split}
  \end{equation*}
  que claramente converge a zero, provando o teorema.
\end{proof}

\begin{corollary}
  A $\mathcal{N}(0,1)$ Ă© a Ășnica distribuiĂ§ĂŁo $\mu \in \mathcal{L}^3$ que possui variĂąncia $1$ e Ă© tal que se $X, Y$ sĂŁo \iid com distribuiĂ§ĂŁo $\mu$, entĂŁo $(X + Y)/\sqrt{2}$ tambĂ©m possuem distribuiĂ§ĂŁo $\mu$.
  Em outras palavras, $\mathcal{N}(0, \sigma^2)$, para $\sigma \geq 0$, sĂŁo os Ășnicos pontos fixos de $\Gamma$ em $\mathcal{L}^3$.
\end{corollary}

\begin{proof}
  Usando a invariĂąncia enunciada acima, temos que
  \begin{equation}
    \frac{X_1 + \dots + X_{2^k}}{\sqrt{2^k}} \distr \mu.
  \end{equation}
  Mas pelo Teorema central do limite, a distribuiĂ§ĂŁo dessa combinaĂ§ĂŁo de $X_i$ deve convergir a $\mathcal{N}(0,1)$, logo temos $\mu = \mathcal{N}(0,1)$.
\end{proof}

   Pois so temos que calcular
   \begin{equation}
   f(\xi):= \frac{1}{\sqrt{2\pi}}\int_{\bbR} e^{-\frac{x^2}{2}} \cos(\xi x) \dd x.
   \end{equation}
 Derivando a funĂ§ĂŁo na integral (o que podemos fazer por que $|x e^{-\frac{x^2}{2}}|$ e integravel) temos
 \begin{equation}
  f'(\xi):=- \frac{1}{\sqrt{2\pi}}\int_{\bbR} e^{-\frac{x^2}{2}} x \sin(\xi x) \dd x.
 \end{equation}
Pois integrando por parte obtemos
 \begin{equation}
  f'(\xi):=- \frac{1}{\sqrt{2\pi}}\int_{\bbR} e^{-\frac{x^2}{2}} \xi \cos(\xi x) \dd x= -\xi f(\xi).
 \end{equation}
   EntĂŁo $f$ e soluĂ§ĂŁo da equaĂ§ĂŁo diferential $f'(\xi)+ \xi f(\xi)=0$, o que, junto com a condiĂ§ĂŁo $f(0)=0$
   da o resultado $f(\xi)= \exp(-\xi^2/2)$.
  \end{proof}


  \begin{theorem}
   A funĂ§ĂŁo caracterĂ­stica $\hat \phi_X$ determina a distribuiĂ§ĂŁo de $X$, o de jeito equivalente,
   a transformaĂ§ĂŁo $\mu \to \hat \mu$ e injectiva (no espaĂ§o das probabilidade em $\bbR^d$).
  \end{theorem}

  \begin{proof}
   Consideramos o caso $d=1$ primeiro. Chamamos $g_{\sigma}$ a densidade da lei gaussiana
   $$ g_{\sigma}(x):=\frac{1}{\sigma \sqrt{2\pi}} e^{-\frac{x^2}{2\sigma^2}}, \qquad x \in \bbR.$$
   Definimos $\mu_{\sigma}$ a convoluĂ§ĂŁo de $\mu$ com $g_{\sigma}$ do jeito seguinte
   \begin{equation}
   \begin{split}
 f_{\sigma}(x)&:= \int_{\bbR} g_{\sigma}(x-y)\mu(\dd y)= g_{\sigma}\ast\mu,\\
 \mu_{\sigma}(x)&:= f_{\sigma}(x) \dd x.
   \end{split}
 \end{equation}
Vamos mostrar primeiramente que para todo $\sigma>0$,  $\mu_{\sigma}$ e determinada por $\hat \mu$, e pois mostraremos que para
toda funĂ§ĂŁo $\varphi$ continua limitada
   \begin{equation}\label{covigi}
  \lim_{\sigma \to 0} \int_{\bbR} \varphi(x)\mu_{\sigma}( \dd x):= \int_{\bbR} \varphi(x)\mu( \dd x ).
   \end{equation}
Pelo primeiro ponto vamos usar o lema precedente que nos diz que
\begin{equation}
 g_{\sigma}(x)= (\sigma \sqrt{2\pi})^{-1} \int_{\bbR} e^{i\xi x} g_{1/\sigma}(\xi) \dd \xi.
\end{equation}
Em consequencia temos como consequencia do teorema de Fubini (verificando integrabilidade)
\begin{equation}\begin{split}
 f_{\sigma}(x)= \int_{\bbR}g_{\sigma}(x-y)\mu(\dd y)&=
 (\sigma \sqrt{2\pi})^{-1} \int_{\bbR}\left(\int_{\bbR} e^{i\xi(y-x)} g_{1/\sigma}(\xi)  \dd \xi\right) \mu(\dd y) \\
 &= (\sigma \sqrt{2\pi})^{-1} \int_{\bbR} g_{1/\sigma}(\xi) e^{-i\xi x} \left(\int_{\bbR} e^{i\xi y}  \mu(\dd y) \right) \dd \xi \\
 &=(\sigma \sqrt{2\pi})^{-1} \int_{\bbR} g_{1/\sigma}(\xi) e^{-i\xi x} \hat\mu(\xi) \dd \xi.
\end{split}
 \end{equation}
Agora mostramos \eqref{covigi}. Temos, usando Fubini que para $\phi$ continua limitada

\begin{multline}
 \int_{\bbR} \varphi(x)\mu_{\sigma}( \dd x)= \int_{\bbR} \varphi(x) \left(\int_{\bbR} g_\sigma(y-x) \mu(\dd y)\right) \dd x \\
   \int_{\bbR} \left( \int_{\bbR} \varphi(x)  g_\sigma(y-x) \dd x \right) \mu(\dd y)=  \int_{\bbR}  \varphi\ast g_\sigma(y) \mu(\dd y).
 \end{multline}
Pois, usando
\begin{equation}
 \int_{\bbR} g_\sigma(x) \dd x \quad \text{ and } \quad \forall \gep>0  \, \lim_{\sigma\to 0} \int_{\{ |x|>\gep \}} g_\sigma(x) \dd x=0,
\end{equation}
e facil mostrar que para todo $y$
\begin{equation}
  \lim_{\sigma \to 0} \varphi\ast g_\sigma(y)=\varphi(y).
\end{equation}
Pois concluimos por convergĂȘncia dominada (lembra que $\phi$ e limitada e $|\varphi\ast g_\sigma(x)|\le \sup |\phi|$) que \eqref{covigi} vale.

\medskip

O caso geral $d\ge 1$ e similar, trocando $g_{\sigma}$ pela seguinte versĂŁo multidimensional.
$$g_{\sigma}^{(d)}(x):=\prod_{i=1}^d g_{\sigma}(x_i).$$
\end{proof}


Acabamos essa introduĂ§ĂŁo a funĂ§oes caracterĂ­stica com a observaĂ§ĂŁo seguinte

\begin{proposition}\label{prop:taylorcara}
 SejĂ  $X=(X_1,\dots,X_d)$ um vetor aleatĂłrio em $\cL^2(\bbR^d)$ (todas coordenadas possuem segundo momento). EntĂŁo $\hat \phi_X(\xi)$ e $C^2$ e
 temos na vizinhanĂ§a de zero
 \begin{equation}\label{taylorblabla}
  \hat \phi_X(\xi)=1+i \sum_{j=1}^d \xi_j E[X_j]- \frac{1}{2} \sum_{j=1}^d\sum_{k=1}^d \xi_j \xi_k E[X_j X_j]+ o(|\xi|^2).
 \end{equation}
\end{proposition}

  \begin{proof}
   Como todos as coordenadas ficam em $\cL^2$ podemos usar teoremas usuais de derivaĂ§ĂŁo de integrais  para obter que
   \begin{equation}
    \frac{\partial  \hat \phi_X}{\partial \xi_i}(\xi)=  iE[X_i e^{\xi.X}] \quad \text{ and }  \frac{\partial^2  \hat \phi_X}{\partial \xi_j \partial \xi_k}(\xi)
    \hat \phi_X(\xi)=-E[X_jX_k e^{\xi.X}].
   \end{equation}
   que sĂŁo funĂ§oes continuas por convergĂȘncia dominada.
O segundo resultado segue da formula de Taylor applicada em zero.
  \end{proof}
Observamos que o resultado acima pode ser extendido no caso $\cL^p$ mas o caso $\cL^2$ e de maior importancia para nosso objetivo.

\section{O Teorema de Levy}

O Teorema de Levy nos leva uma conexĂŁo entre a noĂ§ĂŁo de funĂ§ĂŁo caracterĂ­stica e a convergĂȘncia fraca.


\begin{theorem}
 Uma sequencia de distribuiĂ§ĂŁo $\mu_n$ converge fracamente para $\mu$ se e sĂČ se
 \begin{equation}
  \forall \xi \in \bbR^d, \quad \lim_{n\to \infty}\hat \mu_n(\xi)=\hat \mu(\xi).
 \end{equation}
\end{theorem}


\begin{proof}
 Uma implicaĂ§ĂŁo do teorema segue imediatamente da definiĂ§ĂŁo da convergĂȘncia fraca pois
 $x \to e^{x.\xi}$ e continua e limitada entĂŁo
 \begin{equation}
 \lim_{n\to \infty}\int_{\bbR^d} e^{\xi.x} \mu_n(\dd x)=\int_{\bbR^d} e^{\xi.x} \mu(\dd x).
 \end{equation}

 \medskip

 Mostramos agora a recĂ­proca no caso $d=1$ (o caso geral $d\ge 1$ segue de jeito analogo).
 Sendo $f\in C^c(\bbR)$ e $g_{\sigma}$ definido por
 $$ g_{\sigma}(x)=(\sigma \sqrt{2\pi})^{-1} \exp\left(-\frac{x^2}{2\sigma^2}\right) $$
 reparamos que $f\ast g_{\sigma}$ converge uniformemente por $f$ quando $\sigma \to 0$, pois
 $f$ sendo absolutamente continua temos
 \begin{multline}
  \left |\int f(x-y) g_{\sigma}(y) \dd x -f(x) \right|\\
  =   |\int (f(x-\sigma y)-f(x)) g_{1}(y) \dd x|
  \le \int \go_f(\sigma |y|) g_{1}(y) \dd x
 \end{multline}
onde $\go_f(t):= \max_{|x-y|\le t} |f(x)-f(y)|$, e podemos concluir por convergĂȘncia dominada.
EntĂŁo pela ProposiĂ§ĂŁo \ref{prop:dafuncao} vai ser sufficiente de verificar a convergĂȘncia para
$$H:= \{ \varphi=f \ast g_{\sigma}\ : \ \sigma>0, \ f\in C^c(\bbR)  \}$$
 que fica denso em $C^c(\bbR^d)$.

 \medskip

 A computaĂ§ĂŁo para provar a injectividade da transformada de Fourier, temos para qualquer probabilidade $\nu$
 \begin{multline}
  \int f \ast g_{\sigma} \ \dd \nu= \int f(x) g_{\sigma} \ast \nu(x) \dd x\\
  =\int f(x)  \left( (\sqrt{2\pi} \sigma)^{-1}\int e^{-i\xi x}g_{1/\sigma}(\xi) \hat \nu(\xi)
  \dd
  \xi \right) \dd x.
 \end{multline}
Como $\hat \mu_n(\xi) \to \hat \mu(\xi)$, por convergĂȘncia dominada temos para todo $x$
 $$\int e^{-i\xi x}g_{1/\sigma}(\xi) \hat \mu_n(\xi)
  \dd
  \xi  \to \int e^{-i\xi x}g_{1/\sigma}(\xi) \hat \mu(\xi)
  \dd
  \xi.$$
  Usando de novo convergĂȘncia dominada (as integrais sĂŁo acima sĂŁo menor que $1$ em valor absoluta) pela integral em $x$, concluimos que
  \begin{equation}
    \lim_{n\to \infty} \int f \ast g_{\sigma} \dd \nu_n = \int f \ast g_{\sigma} \dd \nu.
  \end{equation}
O que conclui a demostraĂ§ĂŁo.
\end{proof}

\subsection{O teorema central do limite em $\bbR$}

Agora estamos pronto para mostrar
\begin{theorem}
 Seja $(X_n)_{n\ge 1}$ uma sequencia uma sequencia de variĂĄveis aleatĂłrias em $\cL^2$, de varianca $\sigma^2$.
 EntĂŁo temos a seguinte convergĂȘncia em distribuiĂ§ĂŁo

 \begin{equation}
  \frac{1}{\sqrt{n}}\left(X_1+\dots+X_n-n E[X_1]\right) \Rightarrow \cN(0,\sigma^2)
 \end{equation}

 De jeito equivalente temos para todos reais $a<b$

 \begin{equation}
 \lim_{n\to \infty} P\left[  X_1+\dots+X_n\in [nE[X_1]+\sqrt{n}a \ ,\  nE[X_1]+\sqrt{n}b]\right]=\frac{1}{\sigma \sqrt{2\pi}} \int^b_a e^{\frac{x^2}{2\sigma^2}} \dd x.
 \end{equation}


\end{theorem}

\begin{proof}
 Pelo teorema de Levy, e sufficiente de mostrar convergĂȘncia pontual das funĂ§oes caracteristica.
 Nota que podemos se satisfazer da prova quando $E[X_1]=0$ substituindo $X$ por $X-E[X]$ se for necessario.
 Definimos $Z_n:= \frac{1}{\sqrt{n}}\left(X_1+\dots+X_n\right).$
 Temos

 \begin{multline}
  \hat \phi_{Z_n}(\xi)=E\left[\exp\left( \frac{i\xi}{\sqrt{n}}\left(X_1+\dots+X_n\right) \right)  \right]\\=
  E\left[ \exp\left( \frac{i\xi X_1}{\sqrt{n}} \right) \right]^{n}=  \hat \phi_{X_1}(n^{-1/2}\xi)^n.
 \end{multline}
Sabemos da ProposiĂ§ĂŁo \ref{prop:taylorcara} que quando $\xi\to 0$
\begin{equation}
 \hat \phi_{X_1}(\xi)= 1-\frac{\xi^2\sigma^2}{2}+o(\xi^2),
\end{equation}
entĂŁo temos par $\xi$ fixo
\begin{equation}
  \hat \phi_{X_1}(n^{-1/2}\xi)=1- \frac{\xi^2\sigma^2}{2n}+o(n^{-1}),
\end{equation}
e podemos concluir que
\begin{equation}
 \lim_{n\to \infty} \hat \phi_{X_1}(n^{-1/2}\xi)^n= e^{-\frac{\sigma^2\xi^2}{2}}.
\end{equation}
Como sabemos que essa ultima quantidade e a funĂ§ĂŁo caracteristica de $\cN(0,\sigma^2)$ isso acaba a demostraĂ§ĂŁo.

 \medskip

 A formulaĂ§ĂŁo equivalente pode ser obtida aproximando $\1_{[a,b]}$ com funĂ§Ă”es continua (isso funciona por que a distribuiĂ§ĂŁo Gaussiana nĂŁo possui cojuntos unidade tendo probabilidade positiva).

\end{proof}





No caso especial em que $E = \mathbb{R}$, temos vĂĄrios outras maneiras de caracterizar convergĂȘncia em distribuiĂ§ĂŁo.
A primeira Ă© dada pela seguinte

\begin{proposition}
  \label{p:conv_distr_suave}
  Se $\int g \d \mu_n$ converge para $\int g \d \mu$ para toda $g \in C^3$ limitada e com as trĂȘs primeiras derivadas limitadas, entĂŁo $\mu_n \Rightarrow \mu$.
\end{proposition}

\begin{proof}
  Primeiramente, vamos ver que podemos nos concentrar em um conjunto compacto da reta.

  Para isso fixe um $\varepsilon > 0$ e tome $M'$ tal que $\mu\big( [-M', M'] \big) > 1 - \varepsilon / 3$.
  Tomando uma funĂ§ĂŁo $g$ satisfazendo as hipĂłteses do teorema e tal que
  \begin{equation}
    \1{[-M',M']} \leq g \leq \1{[-M'-1,M'+1]},
  \end{equation}
  concluimos que
  \begin{equation}
    \mu_n \big( [-M'-1, M'+1] \big) \geq 1 - \varepsilon/2,
  \end{equation}
  para todo $n$ suficientemente grande.
  Se tomamos $M \geq M'$ suficientemente grande, podemos obter a cota acima para todo $n$ (com $M$ no lugar de $M'+1$ e $\varepsilon$ no lugar de $\varepsilon/2$).

  Fixamos agora uma $f: \mathbb{R} \to \mathbb{R}$ contĂ­nua e limitada.
  Sabemos que Ă© possĂ­vel aproximar $f$ por uma funĂ§ĂŁo $g \in C^3$ de suporte compacto, com $\lVert g \rVert_\infty \leq 2 \lVert f \rVert_\infty$ e $|g - f| \leq \varepsilon/M$ uniformemente no intervalo $[-M,M]$.
  Essa $g$ certamente satisfaz as hipĂłteses do teorema.

  Portanto,
  \begin{equation*}
    \begin{split}
      \Big| \int f \d \mu_n - \int f \d \mu\Big| & \leq 2 \varepsilon \lVert f \rVert_\infty + \Big| \int_{-M}^M f \d \mu_n - \int_{-M}^M f \d \mu\Big|\\
      & \leq 2 \varepsilon \lVert f \rVert_\infty + \frac \varepsilon{M} 2 M + \Big| \int_{-M}^M g \d \mu_n - \int_{-M}^M g \d \mu\Big|\\
      & \leq 2 \varepsilon \lVert f \rVert_\infty + 2 \varepsilon + \Big| \int g \d \mu_n - \int \d \mu\Big|.
    \end{split}
  \end{equation*}
  Como o Ășltimo termo converge a zero e $\varepsilon$ foi escolhido arbitrariamente, isso conclui a prova da proposiĂ§ĂŁo.
\end{proof}

\subsection{O Teorema Central do Limite em $\bbR^d$}

Para formular o teorema central do limite em dimenĂ§ĂŁo superior precisamos introduzir umas noĂ§Ă”es.
Se $X=(X^{(1)},\dots,X^{(d)})$ e um veitor aleatĂłrio em $\cL^1$. Definimos a esperanĂ§a de $X$ como seguinte
$$E[X]= (E[X^{(1)}],\dots,E[X^{(d)}]).$$


\begin{definition}
 Uma veitor aleatĂłrio $(X^{(1)},\dots X^{(d)})$ e chamado de veito gaussiano, se para qualquer $(\gl_1,\dots,\gl_d)$
 a distribuiĂ§ĂŁo de
 $\gl_1 X^{(1)} +\dots+\gl_d X^{(d)}$.
\end{definition}

\begin{exercise}
 Mostrar que se $X=(X^{(1)},\dots X^{(d)})$ e um veitor gaussiano entĂŁo $X^{(1)},\dots,X^{(d)}$ sĂŁo variĂĄveis gaussiana, mas que a reciproca e falsa quando $d\ge 2$.
\end{exercise}


Fica claro que se para uma sequĂȘncia de veitores $(X_n)_{n\ge 1}$ IID em $\cL^2$ temos
$$ \frac{1}{\sqrt{n}}( (X_1+\dots+X_n) -nE[X_1])\ \Longrightarrow \ X, $$
o veitor $X$ precisa ser Gaussiano. Para ver isso e sufficiente aplicar o teorema central limite para
$\gl_1 X^{(1)} +\dots+\gl_d X^d.$

\medskip

Uma segunda observaĂ§ĂŁo e que a distribuiĂ§ĂŁ de um veitor gaussiano e totalmente definida por sua media
$E[X]$
a sua matrice de covarianĂ§a
\begin{equation}

\end{equation}
Para ver isso consideramos o caso $E[X]=0$ (o outro caso sendo similar)
e sendo $X$ um veitor gaussiano cuja matrice de covarianĂ§a $\Sigma$ calculamos a funĂ§ĂŁo caracteristica.
Por definiĂ§ĂŁo, $\xi.X$ sendo uma combinaĂ§ĂŁo linear das coordenadas, tem uma distribuiĂ§ĂŁo Gaussiana.
Pois temos
\begin{equation}
\Var(\xi.X)=\sum_{i,j=1}^d \sigma(i,j) \xi_i \xi_j= (\xi. \Sigma \xi).
\end{equation}
EntĂŁo
\begin{equation}
 \hat \phi_X(\xi)= E[e^{i\xi.X}]=\exp\left(-\frac{(\xi. \Sigma \xi)}{2}\right).
\end{equation}

No caso onde $\Sigma$ a a matrice identidade, e facil de ver que o veitor gaussiano correspondente pode ser obtido
considerando $X^{(1)},\dots,X^{(d)}$ sendo variĂĄveis gaussianas independente de varianĂ§a $1$. Dai podemos costruir um vector gaussiano para $\Sigma$ arbitrario.


\begin{proposition}
Para toda matrix $\Sigma$ simetrica positiva, existe um veitor gaussiano de matrice de covarianĂ§a $\Sigma$.
\end{proposition}
Em geral escrevemos $\cN(m,\Sigma)$ pela distribuiĂ§ĂŁo do veito gaussiano de matrice de covarianĂ§a $\Sigma$ e de media (esperanĂ§a) $m\in \bbR^d$.

\begin{proof}
 Seja $M$ a matrice simetrica positiva tal que $M^2=\Sigma$ ($M=\sqrt{\Sigma}$), e $Z$ um veitor gaussiano de distribuiĂ§ĂŁo $\cN(m,\Sigma)$.
 Definimos $X=MZ$ e verificamos que
 \begin{equation}
 E[X^{(i)}X^{(j)}]=E[ \left(\sum_{k=1}^d m_{i,k} Z^{(k)}\right)\left(\sum_{k=1}^d m_{j,k} Z^{(k)}\right)]= \sum_{k=1}^d m_{i,k}m_{j,k}= \sigma(i,j).
\end{equation}
 \end{proof}


\begin{theorem}
 Seja $(X_n)_{n\ge 1}$ uma sequencia de veitores aleatĂłries em $\cL^2$, de matriz de covarianca $\Sigma$.
 EntĂŁo temos a seguinte convergĂȘncia em distribuiĂ§ĂŁo

 \begin{equation}
  \frac{1}{\sqrt{n}}\left(X_1+\dots+X_n-n E[X_1]\right) \Rightarrow \cN(0.\Sigma).
 \end{equation}

\end{theorem}
