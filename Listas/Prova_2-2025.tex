\noindent {\bf 1)} Sejam $X_1, X_2, \dots$ variĂˇveis aleatĂłrias independentes com distribuiĂ§ĂŁo Poisson de parĂ˘metros $\lambda_1, \lambda_2, \dots$.

\noindent {\bf 2)} Sejam $X_1, X_2, \dots$ variĂˇveis aleatĂłrias i.i.d. e defina
\begin{equation}

\end{equation}
\begin{enumerate}[\quad a)]
\item (Propriedade de Markov) Mostre que para quaisquer $k, n \geq 1$ e qualquer funĂ§ĂŁo $f: \mathbb{R}^k \to \mathbb{R}_+$ limitada,

    = E \big[ f(S_{n + 1}, \dots, S_{n + k}) | S_n \big].

\item (Martingal com respeito Ă  filtraĂ§ĂŁo canĂ´nica) Mostre que se $X_i \in \mathcal{L}^1$ entĂŁo, para qualquer $n \geq 1$,

  \end{equation}
\end{enumerate}

\noindent {\bf 3)} Considere $f: \mathbb{R} \to \mathbb{R}_+$ contĂ­nua e de suporte compacto, que determine uma densidade na reta, ou seja, tal que $\int f(x) dx = 1$.

Para isso, fixamos uma funĂ§ĂŁo $k: \mathbb{R} \to \mathbb{R}_+$ contĂ­nua e de suporte compacto, tal que $\int k(u) du = 1$ e, fixado $\theta > 0$, definimos o estimador
\begin{equation}
  F_\theta^N(x) := \frac{1}{N} \sum_{i = 1}^N \frac{1}{\theta}

\end{equation}

Nosso objetivo Ă© estimar, para $x$ fixado,
\begin{equation}

\end{equation}
\begin{enumerate}[\quad a)]
\item Mostre que
  \begin{equation}
    P \big[ \big| F_\theta^N(x) - f(x) \big| \geq t \big]
    \leq \frac{E \big( ( F_\theta^N(x) - f(x) )^2 \big)}{t^2}.
  \end{equation}
\item Mostre que
  \begin{equation}
    E \big( ( F_\theta^N(x) - f(x) )^2 \big)
    = \big( E (F_\theta^N(x) - f(x)) \big)^2

  \end{equation}
\item Lembrando que $k$ Ă© contĂ­nua e de suporte compacto e supondo ainda que $\int u k(u) du = 0$ e que $f \in C^1$, mostre que
  \begin{equation}

    \leq \frac{\theta^2}{2} \lVert f'' \rVert_\infty \int u^2 k(u) du.
  \end{equation}
  Dica: desenvolva a sĂ©rie de Taylor para $f$ em torno de $x$.
\item Ainda assumindo que $k$ Ă© contĂ­nua e de suporte compacto, mostre que
